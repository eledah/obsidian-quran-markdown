▶ [[سورة الحاقة]] | [[سورة نوح]] ◀

<br>

##### 1

<span class="ayah">بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ سَأَلَ سَآئِلٌۢ بِعَذَابٍۢ وَاقِعٍۢ</span>
<br><span class="ayah_translation">تقاضاکننده‌ای تقاضای عذابی کرد که واقع شد!</span>

##### 2

<span class="ayah">لِّلْكَٰفِرِينَ لَيْسَ لَهُۥ دَافِعٌۭ</span>
<br><span class="ayah_translation">این عذاب مخصوص کافران است، و هیچ کس نمی‌تواند آن را دفع کند،</span>

##### 3

<span class="ayah">مِّنَ ٱللَّهِ ذِى ٱلْمَعَارِجِ</span>
<br><span class="ayah_translation">از سوی خداوند ذی المعارج [= خداوندی که فرشتگانش بر آسمانها صعود و عروج می‌کنند]!</span>

##### 4

<span class="ayah">تَعْرُجُ ٱلْمَلَٰٓئِكَةُ وَٱلرُّوحُ إِلَيْهِ فِى يَوْمٍۢ كَانَ مِقْدَارُهُۥ خَمْسِينَ أَلْفَ سَنَةٍۢ</span>
<br><span class="ayah_translation">فرشتگان و روح [= فرشته مقرّب خداوند] بسوی او عروج می‌کنند در آن روزی که مقدارش پنجاه هزار سال است!</span>

##### 5

<span class="ayah">فَٱصْبِرْ صَبْرًۭا جَمِيلًا</span>
<br><span class="ayah_translation">پس صبر جمیل پیشه کن،</span>

##### 6

<span class="ayah">إِنَّهُمْ يَرَوْنَهُۥ بَعِيدًۭا</span>
<br><span class="ayah_translation">زیرا آنها آن روز را دور می‌بینند،</span>

##### 7

<span class="ayah">وَنَرَىٰهُ قَرِيبًۭا</span>
<br><span class="ayah_translation">و ما آن را نزدیک می‌بینیم!</span>

##### 8

<span class="ayah">يَوْمَ تَكُونُ ٱلسَّمَآءُ كَٱلْمُهْلِ</span>
<br><span class="ayah_translation">همان روز که آسمان همچون فلز گداخته می‌شود،</span>

##### 9

<span class="ayah">وَتَكُونُ ٱلْجِبَالُ كَٱلْعِهْنِ</span>
<br><span class="ayah_translation">و کوه‌ها مانند پشم رنگین متلاشی خواهد بود،</span>

##### 10

<span class="ayah">وَلَا يَسْـَٔلُ حَمِيمٌ حَمِيمًۭا</span>
<br><span class="ayah_translation">و هیچ دوست صمیمی سراغ دوستش را نمی‌گیرد!</span>

##### 11

<span class="ayah">يُبَصَّرُونَهُمْ ۚ يَوَدُّ ٱلْمُجْرِمُ لَوْ يَفْتَدِى مِنْ عَذَابِ يَوْمِئِذٍۭ بِبَنِيهِ</span>
<br><span class="ayah_translation">آنها را نشانشان می‌دهند (ولی هر کس گرفتار کار خویشتن است)، چنان است که گنهکار دوست می‌دارد فرزندان خود را در برابر عذاب آن روز فدا کند،</span>

##### 12

<span class="ayah">وَصَٰحِبَتِهِۦ وَأَخِيهِ</span>
<br><span class="ayah_translation">و همسر و برادرش را،</span>

##### 13

<span class="ayah">وَفَصِيلَتِهِ ٱلَّتِى تُـْٔوِيهِ</span>
<br><span class="ayah_translation">و قبیله‌اش را که همیشه از او حمایت می‌کرد،</span>

##### 14

<span class="ayah">وَمَن فِى ٱلْأَرْضِ جَمِيعًۭا ثُمَّ يُنجِيهِ</span>
<br><span class="ayah_translation">و همه مردم روی زمین را تا مایه نجاتش گردند؛</span>

##### 15

<span class="ayah">كَلَّآ ۖ إِنَّهَا لَظَىٰ</span>
<br><span class="ayah_translation">امّا هرگز چنین نیست (که با اینها بتوان نجات یافت، آری) شعله‌های سوزان آتش است،</span>

##### 16

<span class="ayah">نَزَّاعَةًۭ لِّلشَّوَىٰ</span>
<br><span class="ayah_translation">دست و پا و پوست سر را می‌کند و می‌برد!</span>

##### 17

<span class="ayah">تَدْعُوا۟ مَنْ أَدْبَرَ وَتَوَلَّىٰ</span>
<br><span class="ayah_translation">و کسانی را که به فرمان خدا پشت کردند صدا می‌زند،</span>

##### 18

<span class="ayah">وَجَمَعَ فَأَوْعَىٰٓ</span>
<br><span class="ayah_translation">و (همچنین آنها که) اموال را جمع و ذخیره کردند!</span>

##### 19

<span class="ayah">۞ إِنَّ ٱلْإِنسَٰنَ خُلِقَ هَلُوعًا</span>
<br><span class="ayah_translation">به یقین انسان حریص و کم‌طاقت آفریده شده است،</span>

##### 20

<span class="ayah">إِذَا مَسَّهُ ٱلشَّرُّ جَزُوعًۭا</span>
<br><span class="ayah_translation">هنگامی که بدی به او رسد بیتابی می‌کند،</span>

##### 21

<span class="ayah">وَإِذَا مَسَّهُ ٱلْخَيْرُ مَنُوعًا</span>
<br><span class="ayah_translation">و هنگامی که خوبی به او رسد مانع دیگران می‌شود (و بخل می‌ورزد)،</span>

##### 22

<span class="ayah">إِلَّا ٱلْمُصَلِّينَ</span>
<br><span class="ayah_translation">مگر نمازگزاران،</span>

##### 23

<span class="ayah">ٱلَّذِينَ هُمْ عَلَىٰ صَلَاتِهِمْ دَآئِمُونَ</span>
<br><span class="ayah_translation">آنها که نمازها را پیوسته بجا می‌آورند،</span>

##### 24

<span class="ayah">وَٱلَّذِينَ فِىٓ أَمْوَٰلِهِمْ حَقٌّۭ مَّعْلُومٌۭ</span>
<br><span class="ayah_translation">و آنها که در اموالشان حق معلومی است...</span>

##### 25

<span class="ayah">لِّلسَّآئِلِ وَٱلْمَحْرُومِ</span>
<br><span class="ayah_translation">برای تقاضاکننده و محروم،</span>

##### 26

<span class="ayah">وَٱلَّذِينَ يُصَدِّقُونَ بِيَوْمِ ٱلدِّينِ</span>
<br><span class="ayah_translation">و آنها که به روز جزا ایمان دارند،</span>

##### 27

<span class="ayah">وَٱلَّذِينَ هُم مِّنْ عَذَابِ رَبِّهِم مُّشْفِقُونَ</span>
<br><span class="ayah_translation">و آنها که از عذاب پروردگارشان بیمناکند،</span>

##### 28

<span class="ayah">إِنَّ عَذَابَ رَبِّهِمْ غَيْرُ مَأْمُونٍۢ</span>
<br><span class="ayah_translation">چرا که هیچ کس از عذاب پروردگارش در امان نیست،</span>

##### 29

<span class="ayah">وَٱلَّذِينَ هُمْ لِفُرُوجِهِمْ حَٰفِظُونَ</span>
<br><span class="ayah_translation">و آنها که دامان خویش را (از بی‌عفّتی) حفظ می‌کنند،</span>

##### 30

<span class="ayah">إِلَّا عَلَىٰٓ أَزْوَٰجِهِمْ أَوْ مَا مَلَكَتْ أَيْمَٰنُهُمْ فَإِنَّهُمْ غَيْرُ مَلُومِينَ</span>
<br><span class="ayah_translation">جز با همسران و کنیزان (که در حکم همسرند آمیزش ندارند)، چرا که در بهره‌گیری از اینها مورد سرزنش نخواهند بود!</span>

##### 31

<span class="ayah">فَمَنِ ٱبْتَغَىٰ وَرَآءَ ذَٰلِكَ فَأُو۟لَٰٓئِكَ هُمُ ٱلْعَادُونَ</span>
<br><span class="ayah_translation">و هر کس جز اینها را طلب کند، متجاوز است!</span>

##### 32

<span class="ayah">وَٱلَّذِينَ هُمْ لِأَمَٰنَٰتِهِمْ وَعَهْدِهِمْ رَٰعُونَ</span>
<br><span class="ayah_translation">و آنها که امانتها و عهد خود را رعایت می‌کنند،</span>

##### 33

<span class="ayah">وَٱلَّذِينَ هُم بِشَهَٰدَٰتِهِمْ قَآئِمُونَ</span>
<br><span class="ayah_translation">و آنها که با ادای شهادتشان قیام می‌نمایند،</span>

##### 34

<span class="ayah">وَٱلَّذِينَ هُمْ عَلَىٰ صَلَاتِهِمْ يُحَافِظُونَ</span>
<br><span class="ayah_translation">و آنها که بر نماز مواظبت دارند،</span>

##### 35

<span class="ayah">أُو۟لَٰٓئِكَ فِى جَنَّٰتٍۢ مُّكْرَمُونَ</span>
<br><span class="ayah_translation">آنان در باغهای بهشتی (پذیرایی و) گرامی داشته می‌شوند.</span>

##### 36

<span class="ayah">فَمَالِ ٱلَّذِينَ كَفَرُوا۟ قِبَلَكَ مُهْطِعِينَ</span>
<br><span class="ayah_translation">این کافران را چه می‌شود که با سرعت نزد تو می‌آیند...</span>

##### 37

<span class="ayah">عَنِ ٱلْيَمِينِ وَعَنِ ٱلشِّمَالِ عِزِينَ</span>
<br><span class="ayah_translation">از راست و چپ، گروه گروه (و آرزوی بهشت دارند)!</span>

##### 38

<span class="ayah">أَيَطْمَعُ كُلُّ ٱمْرِئٍۢ مِّنْهُمْ أَن يُدْخَلَ جَنَّةَ نَعِيمٍۢ</span>
<br><span class="ayah_translation">آیا هر یک از آنها (با این اعمال زشتش) طمع دارد که او را در بهشت پر نعمت الهی وارد کنند؟!</span>

##### 39

<span class="ayah">كَلَّآ ۖ إِنَّا خَلَقْنَٰهُم مِّمَّا يَعْلَمُونَ</span>
<br><span class="ayah_translation">هرگز چنین نیست؛ ما آنها را از آنچه خودشان می‌دانند آفریده‌ایم!</span>

##### 40

<span class="ayah">فَلَآ أُقْسِمُ بِرَبِّ ٱلْمَشَٰرِقِ وَٱلْمَغَٰرِبِ إِنَّا لَقَٰدِرُونَ</span>
<br><span class="ayah_translation">سوگند به پروردگار مشرقها و مغربها که ما قادریم...</span>

##### 41

<span class="ayah">عَلَىٰٓ أَن نُّبَدِّلَ خَيْرًۭا مِّنْهُمْ وَمَا نَحْنُ بِمَسْبُوقِينَ</span>
<br><span class="ayah_translation">که جای آنان را به کسانی بدهیم که از آنها بهترند؛ و ما هرگز مغلوب نخواهیم شد!</span>

##### 42

<span class="ayah">فَذَرْهُمْ يَخُوضُوا۟ وَيَلْعَبُوا۟ حَتَّىٰ يُلَٰقُوا۟ يَوْمَهُمُ ٱلَّذِى يُوعَدُونَ</span>
<br><span class="ayah_translation">آنان را به حال خود واگذار تا در باطل خود فروروند و بازی کنند تا زمانی که روز موعود خود را ملاقات نمایند!</span>

##### 43

<span class="ayah">يَوْمَ يَخْرُجُونَ مِنَ ٱلْأَجْدَاثِ سِرَاعًۭا كَأَنَّهُمْ إِلَىٰ نُصُبٍۢ يُوفِضُونَ</span>
<br><span class="ayah_translation">همان روز که از قبرها بسرعت خارج می‌شوند، گویی به سوی بتها می‌دوند...</span>

##### 44

<span class="ayah">خَٰشِعَةً أَبْصَٰرُهُمْ تَرْهَقُهُمْ ذِلَّةٌۭ ۚ ذَٰلِكَ ٱلْيَوْمُ ٱلَّذِى كَانُوا۟ يُوعَدُونَ</span>
<br><span class="ayah_translation">در حالی که چشمهایشان از شرم و وحشت به زیر افتاده، و پرده‌ای از ذلّت و خواری آنها را پوشانده است! این همان روزی است که به آنها وعده داده می‌شد!</span>
