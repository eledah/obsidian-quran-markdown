<br>

##### 1

<span class="ayah">بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ ٱلْحَمْدُ لِلَّهِ ٱلَّذِى لَهُۥ مَا فِى ٱلسَّمَٰوَٰتِ وَمَا فِى ٱلْأَرْضِ وَلَهُ ٱلْحَمْدُ فِى ٱلْءَاخِرَةِ ۚ وَهُوَ ٱلْحَكِيمُ ٱلْخَبِيرُ</span>
<br><span class="ayah_translation">حمد (و ستایش) مخصوص خداوندی است که تمام آنچه در آسمانها و زمین است از آن اوست؛ و (نیز) حمد (و سپاس) برای اوست در سرای آخرت؛ و او حکیم و آگاه است.</span>

##### 2

<span class="ayah">يَعْلَمُ مَا يَلِجُ فِى ٱلْأَرْضِ وَمَا يَخْرُجُ مِنْهَا وَمَا يَنزِلُ مِنَ ٱلسَّمَآءِ وَمَا يَعْرُجُ فِيهَا ۚ وَهُوَ ٱلرَّحِيمُ ٱلْغَفُورُ</span>
<br><span class="ayah_translation">آنچه در زمین فرومی‌رود و آنچه را از آن برمی‌آید می‌داند، و (همچنین) آنچه از آسمان نازل می‌شود و آنچه بر آن بالا می‌رود؛ و او مهربان و آمرزنده است.</span>

##### 3

<span class="ayah">وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لَا تَأْتِينَا ٱلسَّاعَةُ ۖ قُلْ بَلَىٰ وَرَبِّى لَتَأْتِيَنَّكُمْ عَٰلِمِ ٱلْغَيْبِ ۖ لَا يَعْزُبُ عَنْهُ مِثْقَالُ ذَرَّةٍۢ فِى ٱلسَّمَٰوَٰتِ وَلَا فِى ٱلْأَرْضِ وَلَآ أَصْغَرُ مِن ذَٰلِكَ وَلَآ أَكْبَرُ إِلَّا فِى كِتَٰبٍۢ مُّبِينٍۢ</span>
<br><span class="ayah_translation">کافران گفتند: «قیامت هرگز به سراغ ما نخواهد آمد!» بگو: «آری به پروردگارم سوگند که به سراغ شما خواهد آمد، خداوندی که از غیب آگاه است و به اندازه سنگینی ذرّه‌ای در آسمانها و زمین از علم او دور نخواهد ماند، و نه کوچکتر از آن و نه بزرگتر، مگر اینکه در کتابی آشکار ثبت است!»</span>

##### 4

<span class="ayah">لِّيَجْزِىَ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ ۚ أُو۟لَٰٓئِكَ لَهُم مَّغْفِرَةٌۭ وَرِزْقٌۭ كَرِيمٌۭ</span>
<br><span class="ayah_translation">تا کسانی را که ایمان آورده و کارهای شایسته انجام داده‌اند پاداش دهد؛ برای آنان مغفرت و روزی پرارزشی است!</span>

##### 5

<span class="ayah">وَٱلَّذِينَ سَعَوْ فِىٓ ءَايَٰتِنَا مُعَٰجِزِينَ أُو۟لَٰٓئِكَ لَهُمْ عَذَابٌۭ مِّن رِّجْزٍ أَلِيمٌۭ</span>
<br><span class="ayah_translation">و کسانی که سعی در (تکذیب) آیات ما داشتند و گمان کردند می‌توانند از حوزه قدرت ما بگریزند، عذابی بد و دردناک خواهند داشت!</span>

##### 6

<span class="ayah">وَيَرَى ٱلَّذِينَ أُوتُوا۟ ٱلْعِلْمَ ٱلَّذِىٓ أُنزِلَ إِلَيْكَ مِن رَّبِّكَ هُوَ ٱلْحَقَّ وَيَهْدِىٓ إِلَىٰ صِرَٰطِ ٱلْعَزِيزِ ٱلْحَمِيدِ</span>
<br><span class="ayah_translation">کسانی که به ایشان علم داده شده، آنچه را از سوی پروردگارت بر تو نازل شده حق می دانند و به راه خداوند عزیز و حمید هدایت می‌کند.</span>

##### 7

<span class="ayah">وَقَالَ ٱلَّذِينَ كَفَرُوا۟ هَلْ نَدُلُّكُمْ عَلَىٰ رَجُلٍۢ يُنَبِّئُكُمْ إِذَا مُزِّقْتُمْ كُلَّ مُمَزَّقٍ إِنَّكُمْ لَفِى خَلْقٍۢ جَدِيدٍ</span>
<br><span class="ayah_translation">و کافران گفتند: «آیا مردی را به شما نشان دهیم که به شما خبر می‌دهد هنگامی که (مُردید و) سخت از هم متلاشی شدید، (بار دیگر) آفرینش تازه‌ای خواهید یافت؟!</span>

##### 8

<span class="ayah">أَفْتَرَىٰ عَلَى ٱللَّهِ كَذِبًا أَم بِهِۦ جِنَّةٌۢ ۗ بَلِ ٱلَّذِينَ لَا يُؤْمِنُونَ بِٱلْءَاخِرَةِ فِى ٱلْعَذَابِ وَٱلضَّلَٰلِ ٱلْبَعِيدِ</span>
<br><span class="ayah_translation">آیا او بر خدا دروغ بسته یا به نوعی جنون گرفتار است؟!» (چنین نیست)، بلکه کسانی که به آخرت ایمان نمی‌آورند، در عذاب و گمراهی دوری هستند (و نشانه گمراهی آنها همین انکار شدید است).</span>

##### 9

<span class="ayah">أَفَلَمْ يَرَوْا۟ إِلَىٰ مَا بَيْنَ أَيْدِيهِمْ وَمَا خَلْفَهُم مِّنَ ٱلسَّمَآءِ وَٱلْأَرْضِ ۚ إِن نَّشَأْ نَخْسِفْ بِهِمُ ٱلْأَرْضَ أَوْ نُسْقِطْ عَلَيْهِمْ كِسَفًۭا مِّنَ ٱلسَّمَآءِ ۚ إِنَّ فِى ذَٰلِكَ لَءَايَةًۭ لِّكُلِّ عَبْدٍۢ مُّنِيبٍۢ</span>
<br><span class="ayah_translation">آیا به آنچه پیش رو و پشت سر آنان از آسمان و زمین قرار دارد نگاه نکردند (تا به قدرت خدا بر همه چیز واقف شوند)؟! اگر ما بخواهیم آنها را (با یک زمین‌لرزه) در زمین فرومی‌بریم، یا قطعه‌هایی از سنگهای آسمانی را بر آنها فرومی‌ریزیم؛ در این نشانه‌ای است (بر قدرت خدا) برای هر بنده توبه‌کار!</span>

##### 10

<span class="ayah">۞ وَلَقَدْ ءَاتَيْنَا دَاوُۥدَ مِنَّا فَضْلًۭا ۖ يَٰجِبَالُ أَوِّبِى مَعَهُۥ وَٱلطَّيْرَ ۖ وَأَلَنَّا لَهُ ٱلْحَدِيدَ</span>
<br><span class="ayah_translation">و ما به داوود از سوی خود فضیلتی بزرگ بخشیدیم؛ (ما به کوه‌ها و پرندگان گفتیم:) ای کوه‌ها و ای پرندگان! با او هم‌آواز شوید و همراه او تسبیح خدا گویید! و آهن را برای او نرم کردیم.</span>

##### 11

<span class="ayah">أَنِ ٱعْمَلْ سَٰبِغَٰتٍۢ وَقَدِّرْ فِى ٱلسَّرْدِ ۖ وَٱعْمَلُوا۟ صَٰلِحًا ۖ إِنِّى بِمَا تَعْمَلُونَ بَصِيرٌۭ</span>
<br><span class="ayah_translation">(و به او گفتیم:) زره‌های کامل و فراخ بساز، و حلقه‌ها را به اندازه و متناسب کن! و عمل صالح بجا آورید که من به آنچه انجام میدهید بینا هستم!</span>

##### 12

<span class="ayah">وَلِسُلَيْمَٰنَ ٱلرِّيحَ غُدُوُّهَا شَهْرٌۭ وَرَوَاحُهَا شَهْرٌۭ ۖ وَأَسَلْنَا لَهُۥ عَيْنَ ٱلْقِطْرِ ۖ وَمِنَ ٱلْجِنِّ مَن يَعْمَلُ بَيْنَ يَدَيْهِ بِإِذْنِ رَبِّهِۦ ۖ وَمَن يَزِغْ مِنْهُمْ عَنْ أَمْرِنَا نُذِقْهُ مِنْ عَذَابِ ٱلسَّعِيرِ</span>
<br><span class="ayah_translation">و برای سلیمان باد را مسخّر ساختیم که صبحگاهان مسیر یک ماه را می‌پیمود و عصرگاهان مسیر یک ماه را؛ و چشمه مس (مذاب) را برای او روان ساختیم؛ و گروهی از جنّ پیش روی او به اذن پروردگارش کار می‌کردند؛ و هر کدام از آنها که از فرمان ما سرپیچی می‌کرد، او را عذاب آتش سوزان می‌چشاندیم!</span>

##### 13

<span class="ayah">يَعْمَلُونَ لَهُۥ مَا يَشَآءُ مِن مَّحَٰرِيبَ وَتَمَٰثِيلَ وَجِفَانٍۢ كَٱلْجَوَابِ وَقُدُورٍۢ رَّاسِيَٰتٍ ۚ ٱعْمَلُوٓا۟ ءَالَ دَاوُۥدَ شُكْرًۭا ۚ وَقَلِيلٌۭ مِّنْ عِبَادِىَ ٱلشَّكُورُ</span>
<br><span class="ayah_translation">آنها هر چه سلیمان می‌خواست برایش درست می‌کردند: معبدها، تمثالها، ظروف بزرگ غذا همانند حوضها، و دیگهای ثابت (که از بزرگی قابل حمل و نقل نبود؛ و به آنان گفتیم:) ای آل داوود! شکر (این همه نعمت را) بجا آورید؛ ولی عده کمی از بندگان من شکرگزارند!</span>

##### 14

<span class="ayah">فَلَمَّا قَضَيْنَا عَلَيْهِ ٱلْمَوْتَ مَا دَلَّهُمْ عَلَىٰ مَوْتِهِۦٓ إِلَّا دَآبَّةُ ٱلْأَرْضِ تَأْكُلُ مِنسَأَتَهُۥ ۖ فَلَمَّا خَرَّ تَبَيَّنَتِ ٱلْجِنُّ أَن لَّوْ كَانُوا۟ يَعْلَمُونَ ٱلْغَيْبَ مَا لَبِثُوا۟ فِى ٱلْعَذَابِ ٱلْمُهِينِ</span>
<br><span class="ayah_translation">(با این همه جلال و شکوه سلیمان) هنگامی که مرگ را بر او مقرّر داشتیم، کسی آنها را از مرگ وی آگاه نساخت مگر جنبنده زمین [= موریانه‌] که عصای او را می‌خورد (تا شکست و پیکر سلیمان فرو افتاد)؛ هنگامی که بر زمین افتاد جنّیان فهمیدند که اگر از غیب آگاه بودند در عذاب خوارکننده باقی نمی‌ماندند!</span>

##### 15

<span class="ayah">لَقَدْ كَانَ لِسَبَإٍۢ فِى مَسْكَنِهِمْ ءَايَةٌۭ ۖ جَنَّتَانِ عَن يَمِينٍۢ وَشِمَالٍۢ ۖ كُلُوا۟ مِن رِّزْقِ رَبِّكُمْ وَٱشْكُرُوا۟ لَهُۥ ۚ بَلْدَةٌۭ طَيِّبَةٌۭ وَرَبٌّ غَفُورٌۭ</span>
<br><span class="ayah_translation">برای قوم «سبا» در محل سکونتشان نشانه‌ای (از قدرت الهی) بود: دو باغ (بزرگ و گسترده) از راست و چپ (رودخانه عظیم با میوه‌های فراوان؛ و به آنها گفتیم:) از روزی پروردگارتان بخورید و شکر او را بجا آورید؛ شهری است پاک و پاکیزه، و پروردگاری آمرزنده (و مهربان)!</span>

##### 16

<span class="ayah">فَأَعْرَضُوا۟ فَأَرْسَلْنَا عَلَيْهِمْ سَيْلَ ٱلْعَرِمِ وَبَدَّلْنَٰهُم بِجَنَّتَيْهِمْ جَنَّتَيْنِ ذَوَاتَىْ أُكُلٍ خَمْطٍۢ وَأَثْلٍۢ وَشَىْءٍۢ مِّن سِدْرٍۢ قَلِيلٍۢ</span>
<br><span class="ayah_translation">امّا آنها (از خدا) روی‌گردان شدند، و ما سیل ویرانگر را بر آنان فرستادیم، و دو باغ (پربرکت) شان را به دو باغ (بی‌ارزش) با میوه‌های تلخ و درختان شوره گز و اندکی درخت سدر مبدّل ساختیم!</span>

##### 17

<span class="ayah">ذَٰلِكَ جَزَيْنَٰهُم بِمَا كَفَرُوا۟ ۖ وَهَلْ نُجَٰزِىٓ إِلَّا ٱلْكَفُورَ</span>
<br><span class="ayah_translation">این کیفر را بخاطر کفرانشان به آنها دادیم؛ و آیا جز کفران کننده را کیفر می‌دهیم؟!</span>

##### 18

<span class="ayah">وَجَعَلْنَا بَيْنَهُمْ وَبَيْنَ ٱلْقُرَى ٱلَّتِى بَٰرَكْنَا فِيهَا قُرًۭى ظَٰهِرَةًۭ وَقَدَّرْنَا فِيهَا ٱلسَّيْرَ ۖ سِيرُوا۟ فِيهَا لَيَالِىَ وَأَيَّامًا ءَامِنِينَ</span>
<br><span class="ayah_translation">و میان آنها و شهرهایی که برکت داده بودیم، آبادیهای آشکاری قرار دادیم؛ و سفر در میان آنها را بطور متناسب (با فاصله نزدیک) مقرّر داشتیم؛ (و به آنان گفتیم:) شبها و روزها در این آبادیها با ایمنی (کامل) سفر کنید!</span>

##### 19

<span class="ayah">فَقَالُوا۟ رَبَّنَا بَٰعِدْ بَيْنَ أَسْفَارِنَا وَظَلَمُوٓا۟ أَنفُسَهُمْ فَجَعَلْنَٰهُمْ أَحَادِيثَ وَمَزَّقْنَٰهُمْ كُلَّ مُمَزَّقٍ ۚ إِنَّ فِى ذَٰلِكَ لَءَايَٰتٍۢ لِّكُلِّ صَبَّارٍۢ شَكُورٍۢ</span>
<br><span class="ayah_translation">ولی (این ناسپاس مردم) گفتند: «پروردگارا! میان سفرهای ما دوری بیفکن» (تا بینوایان نتوانند دوش به دوش اغنیا سفر کنند! و به این طریق) آنها به خویشتن ستم کردند! و ما آنان را داستانهایی (برای عبرت دیگران) قرار دادیم و جمعیّتشان را متلاشی ساختیم؛ در این ماجرا، نشانه‌های عبرتی برای هر صابر شکرگزار است.</span>

##### 20

<span class="ayah">وَلَقَدْ صَدَّقَ عَلَيْهِمْ إِبْلِيسُ ظَنَّهُۥ فَٱتَّبَعُوهُ إِلَّا فَرِيقًۭا مِّنَ ٱلْمُؤْمِنِينَ</span>
<br><span class="ayah_translation">(آری) بیقین، ابلیس گمان خود را درباره آنها محقّق یافت که همگی از او پیروی کردند جز گروه اندکی از مؤمنان!</span>

##### 21

<span class="ayah">وَمَا كَانَ لَهُۥ عَلَيْهِم مِّن سُلْطَٰنٍ إِلَّا لِنَعْلَمَ مَن يُؤْمِنُ بِٱلْءَاخِرَةِ مِمَّنْ هُوَ مِنْهَا فِى شَكٍّۢ ۗ وَرَبُّكَ عَلَىٰ كُلِّ شَىْءٍ حَفِيظٌۭ</span>
<br><span class="ayah_translation">او سلطه بر آنان نداشت جز برای اینکه مؤمنان به آخرت را از آنها که در شکّ هستند باز شناسیم؛ و پروردگار تو، نگاهبان همه چیز است!</span>

##### 22

<span class="ayah">قُلِ ٱدْعُوا۟ ٱلَّذِينَ زَعَمْتُم مِّن دُونِ ٱللَّهِ ۖ لَا يَمْلِكُونَ مِثْقَالَ ذَرَّةٍۢ فِى ٱلسَّمَٰوَٰتِ وَلَا فِى ٱلْأَرْضِ وَمَا لَهُمْ فِيهِمَا مِن شِرْكٍۢ وَمَا لَهُۥ مِنْهُم مِّن ظَهِيرٍۢ</span>
<br><span class="ayah_translation">بگو: «کسانی را که غیر از خدا (معبود خود) می‌پندارید بخوانید! (آنها هرگز گرهی از کار شما نمی‌گشایند، چرا که) آنها به اندازه ذرّه‌ای در آسمانها و زمین مالک نیستند، و نه در (خلقت و مالکیّت) آنها شریکند، و نه یاور او (در آفرینش) بودند.</span>

##### 23

<span class="ayah">وَلَا تَنفَعُ ٱلشَّفَٰعَةُ عِندَهُۥٓ إِلَّا لِمَنْ أَذِنَ لَهُۥ ۚ حَتَّىٰٓ إِذَا فُزِّعَ عَن قُلُوبِهِمْ قَالُوا۟ مَاذَا قَالَ رَبُّكُمْ ۖ قَالُوا۟ ٱلْحَقَّ ۖ وَهُوَ ٱلْعَلِىُّ ٱلْكَبِيرُ</span>
<br><span class="ayah_translation">هیچ شفاعتی نزد او، جز برای کسانی که اذن داده، سودی ندارد! (در آن روز همه در اضطرابند) تا زمانی که اضطراب از دلهای آنان زایل گردد (و فرمان از ناحیه او صادر شود؛ در این هنگام مجرمان به شفیعان) می‌گویند: «پروردگارتان چه دستوری داده؟» می‌گویند: «حقّ را (بیان کرد و اجازه شفاعت درباره مستحقّان داد)؛ و اوست بلندمقام و بزرگ‌مرتبه!»</span>

##### 24

<span class="ayah">۞ قُلْ مَن يَرْزُقُكُم مِّنَ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۖ قُلِ ٱللَّهُ ۖ وَإِنَّآ أَوْ إِيَّاكُمْ لَعَلَىٰ هُدًى أَوْ فِى ضَلَٰلٍۢ مُّبِينٍۢ</span>
<br><span class="ayah_translation">بگو: «چه کسی شما را از آسمانها و زمین روزی می‌دهد؟» بگو: «اللّه! و ما یا شما بر (طریق) هدایت یا در ضلالت آشکاری هستیم!»</span>

##### 25

<span class="ayah">قُل لَّا تُسْـَٔلُونَ عَمَّآ أَجْرَمْنَا وَلَا نُسْـَٔلُ عَمَّا تَعْمَلُونَ</span>
<br><span class="ayah_translation">بگو: «شما از گناهی که ما کرده‌ایم سؤال نخواهید شد، (همان گونه که) ما در برابر اعمال شما مسؤول نیستیم!»</span>

##### 26

<span class="ayah">قُلْ يَجْمَعُ بَيْنَنَا رَبُّنَا ثُمَّ يَفْتَحُ بَيْنَنَا بِٱلْحَقِّ وَهُوَ ٱلْفَتَّاحُ ٱلْعَلِيمُ</span>
<br><span class="ayah_translation">بگو: «پروردگار ما همه ما را جمع می‌کند، سپس در میان ما بحق داوری می‌نماید (و صفوف مجرمان را از نیکوکاران جدا می‌سازد)، و اوست داور (و جداکننده) آگاه!»</span>

##### 27

<span class="ayah">قُلْ أَرُونِىَ ٱلَّذِينَ أَلْحَقْتُم بِهِۦ شُرَكَآءَ ۖ كَلَّا ۚ بَلْ هُوَ ٱللَّهُ ٱلْعَزِيزُ ٱلْحَكِيمُ</span>
<br><span class="ayah_translation">بگو: «کسانی را که بعنوان شریک به او ملحق ساخته‌اید به من نشان دهید! هرگز چنین نیست! (او شریک و شبیهی ندارد)، بلکه او خداوند عزیز و حکیم است!</span>

##### 28

<span class="ayah">وَمَآ أَرْسَلْنَٰكَ إِلَّا كَآفَّةًۭ لِّلنَّاسِ بَشِيرًۭا وَنَذِيرًۭا وَلَٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ</span>
<br><span class="ayah_translation">و ما تو را جز برای همه مردم نفرستادیم تا (آنها را به پاداشهای الهی) بشارت دهی و (از عذاب او) بترسانی؛ ولی بیشتر مردم نمی‌دانند!</span>

##### 29

<span class="ayah">وَيَقُولُونَ مَتَىٰ هَٰذَا ٱلْوَعْدُ إِن كُنتُمْ صَٰدِقِينَ</span>
<br><span class="ayah_translation">می‌گویند: «اگر راست می‌گویید، این وعده (رستاخیز) کی خواهد بود؟!»</span>

##### 30

<span class="ayah">قُل لَّكُم مِّيعَادُ يَوْمٍۢ لَّا تَسْتَـْٔخِرُونَ عَنْهُ سَاعَةًۭ وَلَا تَسْتَقْدِمُونَ</span>
<br><span class="ayah_translation">بگو: «وعده شما روزی خواهد بود که نه ساعتی از آن تأخیر می‌کنید و نه (بر آن) پیشی خواهید گرفت!»</span>

##### 31

<span class="ayah">وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لَن نُّؤْمِنَ بِهَٰذَا ٱلْقُرْءَانِ وَلَا بِٱلَّذِى بَيْنَ يَدَيْهِ ۗ وَلَوْ تَرَىٰٓ إِذِ ٱلظَّٰلِمُونَ مَوْقُوفُونَ عِندَ رَبِّهِمْ يَرْجِعُ بَعْضُهُمْ إِلَىٰ بَعْضٍ ٱلْقَوْلَ يَقُولُ ٱلَّذِينَ ٱسْتُضْعِفُوا۟ لِلَّذِينَ ٱسْتَكْبَرُوا۟ لَوْلَآ أَنتُمْ لَكُنَّا مُؤْمِنِينَ</span>
<br><span class="ayah_translation">کافران گفتند: «ما هرگز به این قرآن و کتابهای دیگری که پیش از آن بوده ایمان نخواهیم آورد!» اگر ببینی هنگامی که این ستمگران در پیشگاه پروردگارشان (برای حساب و جزا) نگه داشته شده‌اند در حالی که هر کدام گناه خود را به گردن دیگری می‌اندازد (از وضع آنها تعجّب می‌کنی)! مستضعفان به مستکبران می‌گویند: «اگر شما نبودید ما مؤمن بودیم!»</span>

##### 32

<span class="ayah">قَالَ ٱلَّذِينَ ٱسْتَكْبَرُوا۟ لِلَّذِينَ ٱسْتُضْعِفُوٓا۟ أَنَحْنُ صَدَدْنَٰكُمْ عَنِ ٱلْهُدَىٰ بَعْدَ إِذْ جَآءَكُم ۖ بَلْ كُنتُم مُّجْرِمِينَ</span>
<br><span class="ayah_translation">(امّا) مستکبران به مستضعفان پاسخ می‌دهند: «آیا ما شما را از هدایت بازداشتیم بعد از آنکه به سراغ شما آمد (و آن را بخوبی دریافتید)؟! بلکه شما خود مجرم بودید!»</span>

##### 33

<span class="ayah">وَقَالَ ٱلَّذِينَ ٱسْتُضْعِفُوا۟ لِلَّذِينَ ٱسْتَكْبَرُوا۟ بَلْ مَكْرُ ٱلَّيْلِ وَٱلنَّهَارِ إِذْ تَأْمُرُونَنَآ أَن نَّكْفُرَ بِٱللَّهِ وَنَجْعَلَ لَهُۥٓ أَندَادًۭا ۚ وَأَسَرُّوا۟ ٱلنَّدَامَةَ لَمَّا رَأَوُا۟ ٱلْعَذَابَ وَجَعَلْنَا ٱلْأَغْلَٰلَ فِىٓ أَعْنَاقِ ٱلَّذِينَ كَفَرُوا۟ ۚ هَلْ يُجْزَوْنَ إِلَّا مَا كَانُوا۟ يَعْمَلُونَ</span>
<br><span class="ayah_translation">و مستضعفان به مستکبران می‌گویند: «وسوسه‌های فریبکارانه شما در شب و روز (مایه گمراهی ما شد)، هنگامی که به ما دستور می‌دادید که به خداوند کافر شویم و همتایانی برای او قرار دهیم!» و آنان هنگامی که عذاب (الهی) را می‌بینند پشیمانی خود را پنهان می‌کنند (تا بیشتر رسوا نشوند)! و ما غل و زنجیرها در گردن کافران می نهیم؛ آیا جز آنچه عمل می‌کردند به آنها جزا داده می‌شود؟!</span>

##### 34

<span class="ayah">وَمَآ أَرْسَلْنَا فِى قَرْيَةٍۢ مِّن نَّذِيرٍ إِلَّا قَالَ مُتْرَفُوهَآ إِنَّا بِمَآ أُرْسِلْتُم بِهِۦ كَٰفِرُونَ</span>
<br><span class="ayah_translation">و ما در هیچ شهر و دیاری پیامبری بیم‌دهنده نفرستادیم مگر اینکه مترفین آنها (که مست ناز و نعمت بودند) گفتند: «ما به آنچه فرستاده شده‌اید کافریم!»</span>

##### 35

<span class="ayah">وَقَالُوا۟ نَحْنُ أَكْثَرُ أَمْوَٰلًۭا وَأَوْلَٰدًۭا وَمَا نَحْنُ بِمُعَذَّبِينَ</span>
<br><span class="ayah_translation">و گفتند: «اموال و اولاد ما (از همه) بیشتر است (و این نشانه علاقه خدا به ماست!)؛ و ما هرگز مجازات نخواهیم شد!»</span>

##### 36

<span class="ayah">قُلْ إِنَّ رَبِّى يَبْسُطُ ٱلرِّزْقَ لِمَن يَشَآءُ وَيَقْدِرُ وَلَٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ</span>
<br><span class="ayah_translation">بگو: «پروردگار من روزی را برای هر کس بخواهد وسیع یا تنگ می‌کند، (این ربطی به قرب در درگاه او ندارد)؛ ولی بیشتر مردم نمی‌دانند!</span>

##### 37

<span class="ayah">وَمَآ أَمْوَٰلُكُمْ وَلَآ أَوْلَٰدُكُم بِٱلَّتِى تُقَرِّبُكُمْ عِندَنَا زُلْفَىٰٓ إِلَّا مَنْ ءَامَنَ وَعَمِلَ صَٰلِحًۭا فَأُو۟لَٰٓئِكَ لَهُمْ جَزَآءُ ٱلضِّعْفِ بِمَا عَمِلُوا۟ وَهُمْ فِى ٱلْغُرُفَٰتِ ءَامِنُونَ</span>
<br><span class="ayah_translation">اموال و فرزندانتان هرگز شما را نزد ما مقرّب نمی‌سازد، جز کسانی که ایمان بیاورند و عمل صالحی انجام دهند که برای آنان پاداش مضاعف در برابر کارهایی است که انجام داده‌اند؛ و آنها در غرفه‌های (بهشتی) در (نهایت) امنیّت خواهند بود!</span>

##### 38

<span class="ayah">وَٱلَّذِينَ يَسْعَوْنَ فِىٓ ءَايَٰتِنَا مُعَٰجِزِينَ أُو۟لَٰٓئِكَ فِى ٱلْعَذَابِ مُحْضَرُونَ</span>
<br><span class="ayah_translation">و کسانی که برای انکار و ابطال آیات ما تلاش می‌کنند و می‌پندارند از چنگ قدرت ما فرار خواهند کرد، در عذاب (الهی) احضار می‌شوند!</span>

##### 39

<span class="ayah">قُلْ إِنَّ رَبِّى يَبْسُطُ ٱلرِّزْقَ لِمَن يَشَآءُ مِنْ عِبَادِهِۦ وَيَقْدِرُ لَهُۥ ۚ وَمَآ أَنفَقْتُم مِّن شَىْءٍۢ فَهُوَ يُخْلِفُهُۥ ۖ وَهُوَ خَيْرُ ٱلرَّٰزِقِينَ</span>
<br><span class="ayah_translation">بگو: «پروردگارم روزی را برای هر کس بخواهد وسعت می‌بخشد، و برای هر کس بخواهد تنگ (و محدود) می‌سازد؛ و هر چیزی را (در راه او) انفاق کنید، عوض آن را می‌دهد (و جای آن را پر می‌کند)؛ و او بهترین روزی‌دهندگان است!»</span>

##### 40

<span class="ayah">وَيَوْمَ يَحْشُرُهُمْ جَمِيعًۭا ثُمَّ يَقُولُ لِلْمَلَٰٓئِكَةِ أَهَٰٓؤُلَآءِ إِيَّاكُمْ كَانُوا۟ يَعْبُدُونَ</span>
<br><span class="ayah_translation">(به خاطر بیاور) روزی را که خداوند همه آنان را بر می‌انگیزد، سپس به فرشتگان می‌گوید: «آیا اینها شما را پرستش می‌کردند؟!»</span>

##### 41

<span class="ayah">قَالُوا۟ سُبْحَٰنَكَ أَنتَ وَلِيُّنَا مِن دُونِهِم ۖ بَلْ كَانُوا۟ يَعْبُدُونَ ٱلْجِنَّ ۖ أَكْثَرُهُم بِهِم مُّؤْمِنُونَ</span>
<br><span class="ayah_translation">آنها می‌گویند: «منزّهی (از اینکه همتایی داشته باشی)! تنها تو ولیّ مائی، نه آنها؛ (آنها ما را پرستش نمی‌کردند) بلکه جنّ را پرستش می‌نمودند؛ و اکثرشان به آنها ایمان داشتند!»</span>

##### 42

<span class="ayah">فَٱلْيَوْمَ لَا يَمْلِكُ بَعْضُكُمْ لِبَعْضٍۢ نَّفْعًۭا وَلَا ضَرًّۭا وَنَقُولُ لِلَّذِينَ ظَلَمُوا۟ ذُوقُوا۟ عَذَابَ ٱلنَّارِ ٱلَّتِى كُنتُم بِهَا تُكَذِّبُونَ</span>
<br><span class="ayah_translation">(آری) امروز هیچ یک از شما نسبت به دیگری مالک سود و زیانی نیست! و به ظالمان می‌گوییم: «بچشید عذاب آتشی را که تکذیب می‌کردید!»</span>

##### 43

<span class="ayah">وَإِذَا تُتْلَىٰ عَلَيْهِمْ ءَايَٰتُنَا بَيِّنَٰتٍۢ قَالُوا۟ مَا هَٰذَآ إِلَّا رَجُلٌۭ يُرِيدُ أَن يَصُدَّكُمْ عَمَّا كَانَ يَعْبُدُ ءَابَآؤُكُمْ وَقَالُوا۟ مَا هَٰذَآ إِلَّآ إِفْكٌۭ مُّفْتَرًۭى ۚ وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لِلْحَقِّ لَمَّا جَآءَهُمْ إِنْ هَٰذَآ إِلَّا سِحْرٌۭ مُّبِينٌۭ</span>
<br><span class="ayah_translation">و هنگامی که آیات روشنگر ما بر آنان خوانده می‌شود، می‌گویند: «او فقط مردی است که می‌خواهد شما را از آنچه پدرانتان می‌پرستیدند بازدارد!» و می‌گویند: «این جز دروغ بزرگی که (به خدا) بسته شده چیز دیگری نیست!» و کافران هنگامی که حق به سراغشان آمد گفتند: «این، جز افسونی آشکار نیست!»</span>

##### 44

<span class="ayah">وَمَآ ءَاتَيْنَٰهُم مِّن كُتُبٍۢ يَدْرُسُونَهَا ۖ وَمَآ أَرْسَلْنَآ إِلَيْهِمْ قَبْلَكَ مِن نَّذِيرٍۢ</span>
<br><span class="ayah_translation">ما (قبلا) ً چیزی از کتابهای آسمانی را به آنان نداده‌ایم که آن را بخوانند (و به اتکّای آن سخنان تو را تکذیب کنند)، و پیش از تو هیچ بیم‌دهنده [= پیامبری] برای آنان نفرستادیم!</span>

##### 45

<span class="ayah">وَكَذَّبَ ٱلَّذِينَ مِن قَبْلِهِمْ وَمَا بَلَغُوا۟ مِعْشَارَ مَآ ءَاتَيْنَٰهُمْ فَكَذَّبُوا۟ رُسُلِى ۖ فَكَيْفَ كَانَ نَكِيرِ</span>
<br><span class="ayah_translation">کسانی که پیش از آنان بودند (نیز آیات الهی را) تکذیب کردند، در حالی که اینها به یک دهم آنچه به آنان دادیم نمی‌رسند! (آری) آنها رسولان مرا تکذیب کردند؛ پس ببین مجازات من (نسبت به آنها) چگونه بود!</span>

##### 46

<span class="ayah">۞ قُلْ إِنَّمَآ أَعِظُكُم بِوَٰحِدَةٍ ۖ أَن تَقُومُوا۟ لِلَّهِ مَثْنَىٰ وَفُرَٰدَىٰ ثُمَّ تَتَفَكَّرُوا۟ ۚ مَا بِصَاحِبِكُم مِّن جِنَّةٍ ۚ إِنْ هُوَ إِلَّا نَذِيرٌۭ لَّكُم بَيْنَ يَدَىْ عَذَابٍۢ شَدِيدٍۢ</span>
<br><span class="ayah_translation">بگو: «شما را تنها به یک چیز اندرز می‌دهم، و آن اینکه: دو نفر دو نفر یا یک نفر یک نفر برای خدا قیام کنید، سپس بیندیشید این دوست و همنشین شما [= محمّد] هیچ گونه جنونی ندارد؛ او فقط بیم‌دهنده شما در برابر عذاب شدید (الهی) است!»</span>

##### 47

<span class="ayah">قُلْ مَا سَأَلْتُكُم مِّنْ أَجْرٍۢ فَهُوَ لَكُمْ ۖ إِنْ أَجْرِىَ إِلَّا عَلَى ٱللَّهِ ۖ وَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ شَهِيدٌۭ</span>
<br><span class="ayah_translation">بگو: «هر اجر و پاداشی از شما خواسته‌ام برای خود شماست؛ اجر من تنها بر خداوند است، و او بر همه چیز گواه است!»</span>

##### 48

<span class="ayah">قُلْ إِنَّ رَبِّى يَقْذِفُ بِٱلْحَقِّ عَلَّٰمُ ٱلْغُيُوبِ</span>
<br><span class="ayah_translation">بگو: «پروردگار من حق را (بر دل پیامبران خود) می‌افکند، که او دانای غیبها (و اسرار نهان) است.»</span>

##### 49

<span class="ayah">قُلْ جَآءَ ٱلْحَقُّ وَمَا يُبْدِئُ ٱلْبَٰطِلُ وَمَا يُعِيدُ</span>
<br><span class="ayah_translation">بگو: «حق آمد! و باطل (کاری از آن ساخته نیست و) نمی‌تواند آغازگر چیزی باشد و نه تجدیدکننده آن!»</span>

##### 50

<span class="ayah">قُلْ إِن ضَلَلْتُ فَإِنَّمَآ أَضِلُّ عَلَىٰ نَفْسِى ۖ وَإِنِ ٱهْتَدَيْتُ فَبِمَا يُوحِىٓ إِلَىَّ رَبِّىٓ ۚ إِنَّهُۥ سَمِيعٌۭ قَرِيبٌۭ</span>
<br><span class="ayah_translation">بگو: «اگر من گمراه شوم، از ناحیه خود گمراه می‌شوم؛ و اگر هدایت یابم، به وسیله آنچه پروردگارم به من وحی می‌کند هدایت می‌یابم؛ او شنوای نزدیک است!»</span>

##### 51

<span class="ayah">وَلَوْ تَرَىٰٓ إِذْ فَزِعُوا۟ فَلَا فَوْتَ وَأُخِذُوا۟ مِن مَّكَانٍۢ قَرِيبٍۢ</span>
<br><span class="ayah_translation">اگر ببینی هنگامی که فریادشان بلند می‌شود امّا نمی‌توانند (از عذاب الهی) بگریزند، و آنها را از جای نزدیکی (که حتّی انتظارش را ندارند) می‌گیرند (از درماندگی آنها تعجّب خواهی کرد)!</span>

##### 52

<span class="ayah">وَقَالُوٓا۟ ءَامَنَّا بِهِۦ وَأَنَّىٰ لَهُمُ ٱلتَّنَاوُشُ مِن مَّكَانٍۭ بَعِيدٍۢ</span>
<br><span class="ayah_translation">و (در آن حال) می‌گویند: «به حقّ ایمان آوردیم!»، ولی چگونه می‌توانند از فاصله دور به آن دسترسی پیدا کنند!</span>

##### 53

<span class="ayah">وَقَدْ كَفَرُوا۟ بِهِۦ مِن قَبْلُ ۖ وَيَقْذِفُونَ بِٱلْغَيْبِ مِن مَّكَانٍۭ بَعِيدٍۢ</span>
<br><span class="ayah_translation">آنها پیش از این (که در نهایت آزادی بودند) به آن کافر شدند و دورا دور، و غائبانه (و بدون آگاهی). نسبت‌های ناروا می‌دادند.</span>

##### 54

<span class="ayah">وَحِيلَ بَيْنَهُمْ وَبَيْنَ مَا يَشْتَهُونَ كَمَا فُعِلَ بِأَشْيَاعِهِم مِّن قَبْلُ ۚ إِنَّهُمْ كَانُوا۟ فِى شَكٍّۢ مُّرِيبٍۭ</span>
<br><span class="ayah_translation">(سرانجام) میان آنها و خواسته‌هایشان جدایی افکنده شد، همان گونه که با پیروان (و هم‌مسلکان) آنها از قبل عمل شد، چرا که آنها در شک و تردید بودند!</span>
