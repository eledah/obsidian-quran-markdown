##### 4

##### 5
<span class="ayah hovertext" data-hover="و به سقف برافراشته‌">وَٱلسَّقْفِ ٱلْمَرْفُوعِ</span>
##### 6
<span class="ayah hovertext" data-hover="و به درياى سرشار [از آتش‌]">وَٱلْبَحْرِ ٱلْمَسْجُورِ</span>
##### 7
<span class="ayah hovertext" data-hover="كه عذاب پروردگارت واقع شدنى است‌">إِنَّ عَذَابَ رَبِّكَ لَوَٰقِعٌۭ</span>
##### 8
<span class="ayah hovertext" data-hover="و برگردانى ندارد">مَّا لَهُۥ مِن دَافِعٍۢ</span>
##### 9
<span class="ayah hovertext" data-hover="روزى كه آسمان به شدت درهم گردد">يَوْمَ تَمُورُ ٱلسَّمَآءُ مَوْرًۭا</span>
##### 10
<span class="ayah hovertext" data-hover="و كوهها به شدت روان شود">وَتَسِيرُ ٱلْجِبَالُ سَيْرًۭا</span>
##### 11
<span class="ayah hovertext" data-hover="آرى در چنين روزى واى بر دروغ‌انگاران‌">فَوَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ</span>
##### 12
<span class="ayah hovertext" data-hover="كسانى كه به ژاژخايى سرگرمند">ٱلَّذِينَ هُمْ فِى خَوْضٍۢ يَلْعَبُونَ</span>
##### 13
<span class="ayah hovertext" data-hover="روزى كه به سوى آتش جهنم به شدت رانده شوند">يَوْمَ يُدَعُّونَ إِلَىٰ نَارِ جَهَنَّمَ دَعًّا</span>
##### 14
<span class="ayah hovertext" data-hover="[و گويند] اين همان آتشى است كه آن را دروغ مى‌انگاشتيد">هَٰذِهِ ٱلنَّارُ ٱلَّتِى كُنتُم بِهَا تُكَذِّبُونَ</span>
##### 15
<span class="ayah hovertext" data-hover="آيا پس اين جادوست، يا آنكه شما [به چشم بصيرت‌] نمى‌نگريد؟">أَفَسِحْرٌ هَٰذَآ أَمْ أَنتُمْ لَا تُبْصِرُونَ</span>
##### 16
<span class="ayah hovertext" data-hover="وارد آن شويد، و چه صبر كنيد، چه صبر نكنيد، برايتان يكسان است، فقط در برابر آنچه كرده‌ايد جزا مى‌يابيد">ٱصْلَوْهَا فَٱصْبِرُوٓا۟ أَوْ لَا تَصْبِرُوا۟ سَوَآءٌ عَلَيْكُمْ ۖ إِنَّمَا تُجْزَوْنَ مَا كُنتُمْ تَعْمَلُونَ</span>
##### 17
<span class="ayah hovertext" data-hover="پرهيزگاران در بوستانها و ناز و نعمت‌اند">إِنَّ ٱلْمُتَّقِينَ فِى جَنَّٰتٍۢ وَنَعِيمٍۢ</span>
##### 18
<span class="ayah hovertext" data-hover="از آنچه پروردگارشان به آنان ارزانى داشته است، خرمند، و [از اينكه‌] پروردگارشان از عذاب جهنم در امانشان داشته است‌">فَٰكِهِينَ بِمَآ ءَاتَىٰهُمْ رَبُّهُمْ وَوَقَىٰهُمْ رَبُّهُمْ عَذَابَ ٱلْجَحِيمِ</span>
##### 19
<span class="ayah hovertext" data-hover="به خاطر آنچه كرده‌ايد، به گوارايى بخوريد و بياشاميد">كُلُوا۟ وَٱشْرَبُوا۟ هَنِيٓـًٔۢا بِمَا كُنتُمْ تَعْمَلُونَ</span>
##### 20
<span class="ayah hovertext" data-hover="[آنان‌] بر تختهاى رج زده تكيه زده‌اند و آنان را جفت حوريان درشت چشم گردانيم‌">مُتَّكِـِٔينَ عَلَىٰ سُرُرٍۢ مَّصْفُوفَةٍۢ ۖ وَزَوَّجْنَٰهُم بِحُورٍ عِينٍۢ</span>
##### 21
<span class="ayah hovertext" data-hover="و كسانى كه ايمان آورده‌اند و زاد و رودشان در ايمان از ايشان پيروى كرده‌اند، زاد و رودشان را به ايشان ملحق سازيم، و چيزى از [جزاى‌] عملشان نكاهيم، هر انسانى در گرو كار و كردار خويش است‌">وَٱلَّذِينَ ءَامَنُوا۟ وَٱتَّبَعَتْهُمْ ذُرِّيَّتُهُم بِإِيمَٰنٍ أَلْحَقْنَا بِهِمْ ذُرِّيَّتَهُمْ وَمَآ أَلَتْنَٰهُم مِّنْ عَمَلِهِم مِّن شَىْءٍۢ ۚ كُلُّ ٱمْرِئٍۭ بِمَا كَسَبَ رَهِينٌۭ</span>
##### 22
<span class="ayah hovertext" data-hover="و ايشان را پى در پى از ميوه‌ها و گوشت [پرندگانى‌] كه خوش دارند مى‌دهيم‌">وَأَمْدَدْنَٰهُم بِفَٰكِهَةٍۢ وَلَحْمٍۢ مِّمَّا يَشْتَهُونَ</span>
##### 23
<span class="ayah hovertext" data-hover="در آنجا جامى را كه نه مايه بيهوده‌گويى و نه گناه است، از دست هم مى‌گيرند">يَتَنَٰزَعُونَ فِيهَا كَأْسًۭا لَّا لَغْوٌۭ فِيهَا وَلَا تَأْثِيمٌۭ</span>
##### 24
<span class="ayah hovertext" data-hover="و بر گرداگرد آنان جوانان [خدمتكار]شان مى‌گردند كه گويى ايشان مرواريد نهفته‌اند">۞ وَيَطُوفُ عَلَيْهِمْ غِلْمَانٌۭ لَّهُمْ كَأَنَّهُمْ لُؤْلُؤٌۭ مَّكْنُونٌۭ</span>
##### 25
<span class="ayah hovertext" data-hover="و بعضى از آنان به بعضى ديگر به همپرسى روى آورند">وَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٍۢ يَتَسَآءَلُونَ</span>
##### 26
<span class="ayah hovertext" data-hover="گويند ما پيش از اين، در ميان خانواده‌مان بيمناك بوديم‌">قَالُوٓا۟ إِنَّا كُنَّا قَبْلُ فِىٓ أَهْلِنَا مُشْفِقِينَ</span>
##### 27
<span class="ayah hovertext" data-hover="سپس خداوند بر ما منت نهاد [و نعمت داد] و ما را از عذاب آتشباد [جهنم‌] در امان داشت‌">فَمَنَّ ٱللَّهُ عَلَيْنَا وَوَقَىٰنَا عَذَابَ ٱلسَّمُومِ</span>
##### 28
<span class="ayah hovertext" data-hover="ما پيش از اين او را [به دعا و نيايش‌] مى‌خوانديم، اوست كه نيكوكار مهربان است‌">إِنَّا كُنَّا مِن قَبْلُ نَدْعُوهُ ۖ إِنَّهُۥ هُوَ ٱلْبَرُّ ٱلرَّحِيمُ</span>
##### 29
<span class="ayah hovertext" data-hover="پس پند ده كه تو به نعمت پروردگارت نه كاهنى و نه ديوانه‌">فَذَكِّرْ فَمَآ أَنتَ بِنِعْمَتِ رَبِّكَ بِكَاهِنٍۢ وَلَا مَجْنُونٍ</span>
##### 30
<span class="ayah hovertext" data-hover="مگر گويند شاعرى است كه ما در حق او حوادث روزگار [/مرگ‌] را انتظار مى‌كشيم‌">أَمْ يَقُولُونَ شَاعِرٌۭ نَّتَرَبَّصُ بِهِۦ رَيْبَ ٱلْمَنُونِ</span>
##### 31
<span class="ayah hovertext" data-hover="بگو منتظر باشيد كه من هم همراه شما از منتظرانم‌">قُلْ تَرَبَّصُوا۟ فَإِنِّى مَعَكُم مِّنَ ٱلْمُتَرَبِّصِينَ</span>
##### 32
<span class="ayah hovertext" data-hover="يا مگر خردهايشان آنان را به اين امر فرمان مى‌دهد، يا مگر ايشان قومى طغيانگرند">أَمْ تَأْمُرُهُمْ أَحْلَٰمُهُم بِهَٰذَآ ۚ أَمْ هُمْ قَوْمٌۭ طَاغُونَ</span>
##### 33
<span class="ayah hovertext" data-hover="يا گويند آن را از خود بربافته است، حق اين است كه ايمان نمى‌ورزند">أَمْ يَقُولُونَ تَقَوَّلَهُۥ ۚ بَل لَّا يُؤْمِنُونَ</span>
##### 34
<span class="ayah hovertext" data-hover="پس اگر راستگو هستند سخنى مانند آن بياورند">فَلْيَأْتُوا۟ بِحَدِيثٍۢ مِّثْلِهِۦٓ إِن كَانُوا۟ صَٰدِقِينَ</span>
##### 35
<span class="ayah hovertext" data-hover="يا از هيچ خلق شده‌اند، يا آنكه خودشان خالق [خود]اند؟">أَمْ خُلِقُوا۟ مِنْ غَيْرِ شَىْءٍ أَمْ هُمُ ٱلْخَٰلِقُونَ</span>
##### 36
<span class="ayah hovertext" data-hover="يا آسمانها و زمين را آفريده‌اند، حق اين است كه يقين نمى‌ورزند">أَمْ خَلَقُوا۟ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضَ ۚ بَل لَّا يُوقِنُونَ</span>
##### 37
<span class="ayah hovertext" data-hover="يا گنجينه‌هاى پروردگارت نزد آنهاست، يا آنان چيرگانند؟">أَمْ عِندَهُمْ خَزَآئِنُ رَبِّكَ أَمْ هُمُ ٱلْمُصَۣيْطِرُونَ</span>
##### 38
<span class="ayah hovertext" data-hover="يا نردبانى دارند كه [بر آن بالا رفته‌] [اسرار را] با آن مى‌شنوند، در اين صورت شنونده آنان دليلى آشكار بياورد">أَمْ لَهُمْ سُلَّمٌۭ يَسْتَمِعُونَ فِيهِ ۖ فَلْيَأْتِ مُسْتَمِعُهُم بِسُلْطَٰنٍۢ مُّبِينٍ</span>
##### 39
<span class="ayah hovertext" data-hover="يا مگر او [خداوند] را دختران و شما را پسران است؟">أَمْ لَهُ ٱلْبَنَٰتُ وَلَكُمُ ٱلْبَنُونَ</span>
##### 40
<span class="ayah hovertext" data-hover="يا مگر از ايشان مزدى مى‌طلبى، و ايشان از تاوانى گرانبارند؟">أَمْ تَسْـَٔلُهُمْ أَجْرًۭا فَهُم مِّن مَّغْرَمٍۢ مُّثْقَلُونَ</span>
##### 41
<span class="ayah hovertext" data-hover="يا مگر غيب در اختيار آنان است و ايشان [هر چه خواهند] مى‌نويسند">أَمْ عِندَهُمُ ٱلْغَيْبُ فَهُمْ يَكْتُبُونَ</span>
##### 42
<span class="ayah hovertext" data-hover="يا مگر نيرنگى در سر دارند، ولى كافران خود نيرنگ خورده‌اند">أَمْ يُرِيدُونَ كَيْدًۭا ۖ فَٱلَّذِينَ كَفَرُوا۟ هُمُ ٱلْمَكِيدُونَ</span>
##### 43
<span class="ayah hovertext" data-hover="يا مگر ايشان را خدايى غير از خداوند است، منزه است خداوند از شريكى كه مى‌انگارند">أَمْ لَهُمْ إِلَٰهٌ غَيْرُ ٱللَّهِ ۚ سُبْحَٰنَ ٱللَّهِ عَمَّا يُشْرِكُونَ</span>
##### 44
<span class="ayah hovertext" data-hover="و چون پاره‌اى از آسمان را افتاده بينند، گويند ابرى متراكم [و نعمت‌] است‌">وَإِن يَرَوْا۟ كِسْفًۭا مِّنَ ٱلسَّمَآءِ سَاقِطًۭا يَقُولُوا۟ سَحَابٌۭ مَّرْكُومٌۭ</span>
##### 45
<span class="ayah hovertext" data-hover="پس ايشان را رها كن تا روز خاصشان را كه در آن بيهوش كرده شوند، ديدار كنند">فَذَرْهُمْ حَتَّىٰ يُلَٰقُوا۟ يَوْمَهُمُ ٱلَّذِى فِيهِ يُصْعَقُونَ</span>
##### 46
<span class="ayah hovertext" data-hover="روزى كه نيرنگشان به كار آنان نيايد و يارى نيابند">يَوْمَ لَا يُغْنِى عَنْهُمْ كَيْدُهُمْ شَيْـًۭٔا وَلَا هُمْ يُنصَرُونَ</span>
##### 47
<span class="ayah hovertext" data-hover="و براى ستمكاران عذابى جز اين است، ولى بيشترشان نمى‌دانند">وَإِنَّ لِلَّذِينَ ظَلَمُوا۟ عَذَابًۭا دُونَ ذَٰلِكَ وَلَٰكِنَّ أَكْثَرَهُمْ لَا يَعْلَمُونَ</span>
##### 48
<span class="ayah hovertext" data-hover="و تا رسيدن حكم پروردگارت شكيبايى كن، كه تو زير نظر مايى، و چون [از خواب‌] برخاستى سپاسگزارانه پروردگارت را تسبيح گوى‌">وَٱصْبِرْ لِحُكْمِ رَبِّكَ فَإِنَّكَ بِأَعْيُنِنَا ۖ وَسَبِّحْ بِحَمْدِ رَبِّكَ حِينَ تَقُومُ</span>
##### 49
<span class="ayah hovertext" data-hover="و در شب و به هنگام غائب شدن ستارگان نيز او را نيايش كن‌">وَمِنَ ٱلَّيْلِ فَسَبِّحْهُ وَإِدْبَٰرَ ٱلنُّجُومِ</span>
