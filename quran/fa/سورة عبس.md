▶ [[سورة النازعات]] | [[سورة التكوير]] ◀

<br>

##### 1

<span class="ayah">بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ عَبَسَ وَتَوَلَّىٰٓ</span>
<br><span class="ayah_translation">چهره در هم کشید و روی برتافت...</span>

##### 2

<span class="ayah">أَن جَآءَهُ ٱلْأَعْمَىٰ</span>
<br><span class="ayah_translation">از اینکه نابینایی به سراغ او آمده بود!</span>

##### 3

<span class="ayah">وَمَا يُدْرِيكَ لَعَلَّهُۥ يَزَّكَّىٰٓ</span>
<br><span class="ayah_translation">تو چه می‌دانی شاید او پاکی و تقوا پیشه کند،</span>

##### 4

<span class="ayah">أَوْ يَذَّكَّرُ فَتَنفَعَهُ ٱلذِّكْرَىٰٓ</span>
<br><span class="ayah_translation">یا متذکّر گردد و این تذکّر به حال او مفید باشد!</span>

##### 5

<span class="ayah">أَمَّا مَنِ ٱسْتَغْنَىٰ</span>
<br><span class="ayah_translation">امّا آن کس که توانگر است،</span>

##### 6

<span class="ayah">فَأَنتَ لَهُۥ تَصَدَّىٰ</span>
<br><span class="ayah_translation">تو به او روی می‌آوری،</span>

##### 7

<span class="ayah">وَمَا عَلَيْكَ أَلَّا يَزَّكَّىٰ</span>
<br><span class="ayah_translation">در حالی که اگر او خود را پاک نسازد، چیزی بر تو نیست!</span>

##### 8

<span class="ayah">وَأَمَّا مَن جَآءَكَ يَسْعَىٰ</span>
<br><span class="ayah_translation">امّا کسی که به سراغ تو می‌آید و کوشش می‌کند،</span>

##### 9

<span class="ayah">وَهُوَ يَخْشَىٰ</span>
<br><span class="ayah_translation">و از خدا ترسان است،</span>

##### 10

<span class="ayah">فَأَنتَ عَنْهُ تَلَهَّىٰ</span>
<br><span class="ayah_translation">تو از او غافل می‌شوی!</span>

##### 11

<span class="ayah">كَلَّآ إِنَّهَا تَذْكِرَةٌۭ</span>
<br><span class="ayah_translation">هرگز چنین نیست که آنها می‌پندارند؛ این (قرآن) تذکّر و یادآوری است،</span>

##### 12

<span class="ayah">فَمَن شَآءَ ذَكَرَهُۥ</span>
<br><span class="ayah_translation">و هر کس بخواهد از آن پند می‌گیرد!</span>

##### 13

<span class="ayah">فِى صُحُفٍۢ مُّكَرَّمَةٍۢ</span>
<br><span class="ayah_translation">در الواح پرارزشی ثبت است،</span>

##### 14

<span class="ayah">مَّرْفُوعَةٍۢ مُّطَهَّرَةٍۭ</span>
<br><span class="ayah_translation">الواحی والاقدر و پاکیزه،</span>

##### 15

<span class="ayah">بِأَيْدِى سَفَرَةٍۢ</span>
<br><span class="ayah_translation">به دست سفیرانی است‌</span>

##### 16

<span class="ayah">كِرَامٍۭ بَرَرَةٍۢ</span>
<br><span class="ayah_translation">والا مقام و فرمانبردار و نیکوکار!</span>

##### 17

<span class="ayah">قُتِلَ ٱلْإِنسَٰنُ مَآ أَكْفَرَهُۥ</span>
<br><span class="ayah_translation">مرگ بر این انسان، چقدر کافر و ناسپاس است!</span>

##### 18

<span class="ayah">مِنْ أَىِّ شَىْءٍ خَلَقَهُۥ</span>
<br><span class="ayah_translation">(خداوند) او را از چه چیز آفریده است؟!</span>

##### 19

<span class="ayah">مِن نُّطْفَةٍ خَلَقَهُۥ فَقَدَّرَهُۥ</span>
<br><span class="ayah_translation">او را از نطفه ناچیزی آفرید، سپس اندازه‌گیری کرد و موزون ساخت،</span>

##### 20

<span class="ayah">ثُمَّ ٱلسَّبِيلَ يَسَّرَهُۥ</span>
<br><span class="ayah_translation">سپس راه را برای او آسان کرد،</span>

##### 21

<span class="ayah">ثُمَّ أَمَاتَهُۥ فَأَقْبَرَهُۥ</span>
<br><span class="ayah_translation">بعد او را میراند و در قبر پنهان نمود،</span>

##### 22

<span class="ayah">ثُمَّ إِذَا شَآءَ أَنشَرَهُۥ</span>
<br><span class="ayah_translation">سپس هرگاه بخواهد او را زنده می‌کند!</span>

##### 23

<span class="ayah">كَلَّا لَمَّا يَقْضِ مَآ أَمَرَهُۥ</span>
<br><span class="ayah_translation">چنین نیست که او می‌پندارد؛ او هنوز آنچه را (خدا) فرمان داده، اطاعت نکرده است!</span>

##### 24

<span class="ayah">فَلْيَنظُرِ ٱلْإِنسَٰنُ إِلَىٰ طَعَامِهِۦٓ</span>
<br><span class="ayah_translation">انسان باید به غذای خویش (و آفرینش آن) بنگرد!</span>

##### 25

<span class="ayah">أَنَّا صَبَبْنَا ٱلْمَآءَ صَبًّۭا</span>
<br><span class="ayah_translation">ما آب فراوان از آسمان فرو ریختیم،</span>

##### 26

<span class="ayah">ثُمَّ شَقَقْنَا ٱلْأَرْضَ شَقًّۭا</span>
<br><span class="ayah_translation">سپس زمین را از هم شکافتیم،</span>

##### 27

<span class="ayah">فَأَنۢبَتْنَا فِيهَا حَبًّۭا</span>
<br><span class="ayah_translation">و در آن دانه‌های فراوانی رویاندیم،</span>

##### 28

<span class="ayah">وَعِنَبًۭا وَقَضْبًۭا</span>
<br><span class="ayah_translation">و انگور و سبزی بسیار،</span>

##### 29

<span class="ayah">وَزَيْتُونًۭا وَنَخْلًۭا</span>
<br><span class="ayah_translation">و زیتون و نخل فراوان،</span>

##### 30

<span class="ayah">وَحَدَآئِقَ غُلْبًۭا</span>
<br><span class="ayah_translation">و باغهای پردرخت،</span>

##### 31

<span class="ayah">وَفَٰكِهَةًۭ وَأَبًّۭا</span>
<br><span class="ayah_translation">و میوه و چراگاه،</span>

##### 32

<span class="ayah">مَّتَٰعًۭا لَّكُمْ وَلِأَنْعَٰمِكُمْ</span>
<br><span class="ayah_translation">تا وسیله‌ای برای بهره‌گیری شما و چهارپایانتان باشد!</span>

##### 33

<span class="ayah">فَإِذَا جَآءَتِ ٱلصَّآخَّةُ</span>
<br><span class="ayah_translation">هنگامی که آن صدای مهیب [= صیحه رستاخیز] بیاید، (کافران در اندوه عمیقی فرومی‌روند)!</span>

##### 34

<span class="ayah">يَوْمَ يَفِرُّ ٱلْمَرْءُ مِنْ أَخِيهِ</span>
<br><span class="ayah_translation">در آن روز که انسان از برادر خود می‌گریزد،</span>

##### 35

<span class="ayah">وَأُمِّهِۦ وَأَبِيهِ</span>
<br><span class="ayah_translation">و از مادر و پدرش،</span>

##### 36

<span class="ayah">وَصَٰحِبَتِهِۦ وَبَنِيهِ</span>
<br><span class="ayah_translation">و زن و فرزندانش؛</span>

##### 37

<span class="ayah">لِكُلِّ ٱمْرِئٍۢ مِّنْهُمْ يَوْمَئِذٍۢ شَأْنٌۭ يُغْنِيهِ</span>
<br><span class="ayah_translation">در آن روز هر کدام از آنها وضعی دارد که او را کاملاً به خود مشغول می‌سازد!</span>

##### 38

<span class="ayah">وُجُوهٌۭ يَوْمَئِذٍۢ مُّسْفِرَةٌۭ</span>
<br><span class="ayah_translation">چهره‌هائی در آن روز گشاده و نورانی است،</span>

##### 39

<span class="ayah">ضَاحِكَةٌۭ مُّسْتَبْشِرَةٌۭ</span>
<br><span class="ayah_translation">خندان و مسرور است؛</span>

##### 40

<span class="ayah">وَوُجُوهٌۭ يَوْمَئِذٍ عَلَيْهَا غَبَرَةٌۭ</span>
<br><span class="ayah_translation">و صورتهایی در آن روز غبارآلود است،</span>

##### 41

<span class="ayah">تَرْهَقُهَا قَتَرَةٌ</span>
<br><span class="ayah_translation">و دود تاریکی آنها را پوشانده است،</span>

##### 42

<span class="ayah">أُو۟لَٰٓئِكَ هُمُ ٱلْكَفَرَةُ ٱلْفَجَرَةُ</span>
<br><span class="ayah_translation">آنان همان کافران فاجرند!</span>
