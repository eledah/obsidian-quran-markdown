▶ [[سورة الزمر]] | [[سورة فصلت]] ◀

<br>

##### 1

<span class="ayah">بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ حمٓ</span>
<br><span class="ayah_translation">حم.</span>

##### 2

<span class="ayah">تَنزِيلُ ٱلْكِتَٰبِ مِنَ ٱللَّهِ ٱلْعَزِيزِ ٱلْعَلِيمِ</span>
<br><span class="ayah_translation">این کتابی است که از سوی خداوند قادر و دانا نازل شده است.</span>

##### 3

<span class="ayah">غَافِرِ ٱلذَّنۢبِ وَقَابِلِ ٱلتَّوْبِ شَدِيدِ ٱلْعِقَابِ ذِى ٱلطَّوْلِ ۖ لَآ إِلَٰهَ إِلَّا هُوَ ۖ إِلَيْهِ ٱلْمَصِيرُ</span>
<br><span class="ayah_translation">خداوندی که آمرزنده گناه، پذیرنده توبه، دارای مجازات سخت، و صاحب نعمت فراوان است؛ هیچ معبودی جز او نیست؛ و بازگشت (همه شما) تنها بسوی اوست.</span>

##### 4

<span class="ayah">مَا يُجَٰدِلُ فِىٓ ءَايَٰتِ ٱللَّهِ إِلَّا ٱلَّذِينَ كَفَرُوا۟ فَلَا يَغْرُرْكَ تَقَلُّبُهُمْ فِى ٱلْبِلَٰدِ</span>
<br><span class="ayah_translation">تنها کسانی در آیات ما مجادله می‌کنند که (از روی عناد) کافر شده‌اند؛ پس مبادا رفت و آمد آنان در شهرها (و قدرت‌نمایی آنان) تو را بفریبد!</span>

##### 5

<span class="ayah">كَذَّبَتْ قَبْلَهُمْ قَوْمُ نُوحٍۢ وَٱلْأَحْزَابُ مِنۢ بَعْدِهِمْ ۖ وَهَمَّتْ كُلُّ أُمَّةٍۭ بِرَسُولِهِمْ لِيَأْخُذُوهُ ۖ وَجَٰدَلُوا۟ بِٱلْبَٰطِلِ لِيُدْحِضُوا۟ بِهِ ٱلْحَقَّ فَأَخَذْتُهُمْ ۖ فَكَيْفَ كَانَ عِقَابِ</span>
<br><span class="ayah_translation">پیش از آنها قوم نوح و اقوامی که بعد از ایشان بودند (پیامبرانشان را) تکذیب کردند؛ و هر امّتی در پی آن بود که توطئه کند و پیامبرش را بگیرد (و آزار دهد)، و برای محو حق به مجادله باطل دست زدند؛ امّا من آنها را گرفتم (و سخت مجازات کردم)؛ ببین که مجازات من چگونه بود!</span>

##### 6

<span class="ayah">وَكَذَٰلِكَ حَقَّتْ كَلِمَتُ رَبِّكَ عَلَى ٱلَّذِينَ كَفَرُوٓا۟ أَنَّهُمْ أَصْحَٰبُ ٱلنَّارِ</span>
<br><span class="ayah_translation">و این گونه فرمان پروردگارت درباره کسانی که کافر شدند مسلّم شده که آنها همه اهل آتشند!</span>

##### 7

<span class="ayah">ٱلَّذِينَ يَحْمِلُونَ ٱلْعَرْشَ وَمَنْ حَوْلَهُۥ يُسَبِّحُونَ بِحَمْدِ رَبِّهِمْ وَيُؤْمِنُونَ بِهِۦ وَيَسْتَغْفِرُونَ لِلَّذِينَ ءَامَنُوا۟ رَبَّنَا وَسِعْتَ كُلَّ شَىْءٍۢ رَّحْمَةًۭ وَعِلْمًۭا فَٱغْفِرْ لِلَّذِينَ تَابُوا۟ وَٱتَّبَعُوا۟ سَبِيلَكَ وَقِهِمْ عَذَابَ ٱلْجَحِيمِ</span>
<br><span class="ayah_translation">فرشتگانی که حاملان عرشند و آنها که گرداگرد آن (طواف می‌کنند) تسبیح و حمد پروردگارشان را می‌گویند و به او ایمان دارند و برای مؤمنان استغفار می‌کنند (و می گویند:) پروردگارا! رحمت و علم تو همه چیز را فراگرفته است؛ پس کسانی را که توبه کرده و راه تو را پیروی می‌کنند بیامرز، و آنان را از عذاب دوزخ نگاه دار!</span>

##### 8

<span class="ayah">رَبَّنَا وَأَدْخِلْهُمْ جَنَّٰتِ عَدْنٍ ٱلَّتِى وَعَدتَّهُمْ وَمَن صَلَحَ مِنْ ءَابَآئِهِمْ وَأَزْوَٰجِهِمْ وَذُرِّيَّٰتِهِمْ ۚ إِنَّكَ أَنتَ ٱلْعَزِيزُ ٱلْحَكِيمُ</span>
<br><span class="ayah_translation">پروردگارا! آنها را در باغهای جاویدان بهشت که به آنها وعده فرموده‌ای وارد کن، همچنین از پدران و همسران و فرزندانشان هر کدام که صالح بودند، که تو توانا و حکیمی!</span>

##### 9

<span class="ayah">وَقِهِمُ ٱلسَّيِّـَٔاتِ ۚ وَمَن تَقِ ٱلسَّيِّـَٔاتِ يَوْمَئِذٍۢ فَقَدْ رَحِمْتَهُۥ ۚ وَذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْعَظِيمُ</span>
<br><span class="ayah_translation">و آنان را از بدیها نگاه دار، و هر کس را که در آن روز از بدیها نگاه داری، مشمول رحمتت ساخته‌ای؛ و این است همان رستگاری عظیم!</span>

##### 10

<span class="ayah">إِنَّ ٱلَّذِينَ كَفَرُوا۟ يُنَادَوْنَ لَمَقْتُ ٱللَّهِ أَكْبَرُ مِن مَّقْتِكُمْ أَنفُسَكُمْ إِذْ تُدْعَوْنَ إِلَى ٱلْإِيمَٰنِ فَتَكْفُرُونَ</span>
<br><span class="ayah_translation">کسانی را که کافر شدند روز قیامت صدا می‌زنند که عداوت و خشم خداوند نسبت به شما از عداوت و خشم خودتان نسبت به خودتان بیشتر است، چرا که بسوی ایمان دعوت می‌شدید، ولی انکار می‌کردید!</span>

##### 11

<span class="ayah">قَالُوا۟ رَبَّنَآ أَمَتَّنَا ٱثْنَتَيْنِ وَأَحْيَيْتَنَا ٱثْنَتَيْنِ فَٱعْتَرَفْنَا بِذُنُوبِنَا فَهَلْ إِلَىٰ خُرُوجٍۢ مِّن سَبِيلٍۢ</span>
<br><span class="ayah_translation">آنها می‌گویند: «پروردگارا! ما را دو بار میراندی و دو بار زنده کردی؛ اکنون به گناهان خود معترفیم؛ آیا راهی برای خارج شدن (از دوزخ) وجود دارد؟»</span>

##### 12

<span class="ayah">ذَٰلِكُم بِأَنَّهُۥٓ إِذَا دُعِىَ ٱللَّهُ وَحْدَهُۥ كَفَرْتُمْ ۖ وَإِن يُشْرَكْ بِهِۦ تُؤْمِنُوا۟ ۚ فَٱلْحُكْمُ لِلَّهِ ٱلْعَلِىِّ ٱلْكَبِيرِ</span>
<br><span class="ayah_translation">این بخاطر آن است که وقتی خداوند به یگانگی خوانده می‌شد انکار می‌کردید، و اگر برای او همتایی می‌پنداشتند ایمان می‌آوردید؛ اکنون داوری مخصوص خداوند بلندمرتبه و بزرگ است (و شما را مطابق عدل خود کیفر می‌دهد).</span>

##### 13

<span class="ayah">هُوَ ٱلَّذِى يُرِيكُمْ ءَايَٰتِهِۦ وَيُنَزِّلُ لَكُم مِّنَ ٱلسَّمَآءِ رِزْقًۭا ۚ وَمَا يَتَذَكَّرُ إِلَّا مَن يُنِيبُ</span>
<br><span class="ayah_translation">او کسی است که آیات خود را به شما نشان می‌دهد و از آسمان برای شما روزی (با ارزشی) می‌فرستد؛ تنها کسانی متذکّر این حقایق می‌شوند که بسوی خدا باز می‌گردند.</span>

##### 14

<span class="ayah">فَٱدْعُوا۟ ٱللَّهَ مُخْلِصِينَ لَهُ ٱلدِّينَ وَلَوْ كَرِهَ ٱلْكَٰفِرُونَ</span>
<br><span class="ayah_translation">(تنها) خدا را بخوانید و دین خود را برای او خالص کنید، هرچند کافران ناخشنود باشند!</span>

##### 15

<span class="ayah">رَفِيعُ ٱلدَّرَجَٰتِ ذُو ٱلْعَرْشِ يُلْقِى ٱلرُّوحَ مِنْ أَمْرِهِۦ عَلَىٰ مَن يَشَآءُ مِنْ عِبَادِهِۦ لِيُنذِرَ يَوْمَ ٱلتَّلَاقِ</span>
<br><span class="ayah_translation">او درجات (بندگان صالح) را بالا می‌برد، او صاحب عرش است، روح (مقدّس) را به فرمانش بر هر کس از بندگانش که بخواهد القاء می‌کند تا (مردم را) از روز ملاقات [= روز رستاخیز] بیم دهد!</span>

##### 16

<span class="ayah">يَوْمَ هُم بَٰرِزُونَ ۖ لَا يَخْفَىٰ عَلَى ٱللَّهِ مِنْهُمْ شَىْءٌۭ ۚ لِّمَنِ ٱلْمُلْكُ ٱلْيَوْمَ ۖ لِلَّهِ ٱلْوَٰحِدِ ٱلْقَهَّارِ</span>
<br><span class="ayah_translation">روزی که همه آنان آشکار می‌شوند و چیزی از آنها بر خدا پنهان نخواهد ماند؛ (و گفته می‌شود:) حکومت امروز برای کیست؟ برای خداوند یکتای قهّار است!</span>

##### 17

<span class="ayah">ٱلْيَوْمَ تُجْزَىٰ كُلُّ نَفْسٍۭ بِمَا كَسَبَتْ ۚ لَا ظُلْمَ ٱلْيَوْمَ ۚ إِنَّ ٱللَّهَ سَرِيعُ ٱلْحِسَابِ</span>
<br><span class="ayah_translation">امروز هر کس در برابر کاری که انجام داده است پاداش داده می‌شود؛ امروز هیچ ظلمی نیست؛ خداوند سریع الحساب است!</span>

##### 18

<span class="ayah">وَأَنذِرْهُمْ يَوْمَ ٱلْءَازِفَةِ إِذِ ٱلْقُلُوبُ لَدَى ٱلْحَنَاجِرِ كَٰظِمِينَ ۚ مَا لِلظَّٰلِمِينَ مِنْ حَمِيمٍۢ وَلَا شَفِيعٍۢ يُطَاعُ</span>
<br><span class="ayah_translation">و آنها را از روز نزدیک بترسان، هنگامی که از شدّت وحشت دلها به گلوگاه می‌رسد و تمامی وجود آنها مملوّ از اندوه می‌گردد؛ برای ستمکاران دوستی وجود ندارد، و نه شفاعت کننده‌ای که شفاعتش پذیرفته شود.</span>

##### 19

<span class="ayah">يَعْلَمُ خَآئِنَةَ ٱلْأَعْيُنِ وَمَا تُخْفِى ٱلصُّدُورُ</span>
<br><span class="ayah_translation">او چشمهایی را که به خیانت می‌گردد و آنچه را سینه‌ها پنهان می‌دارند، می‌داند.</span>

##### 20

<span class="ayah">وَٱللَّهُ يَقْضِى بِٱلْحَقِّ ۖ وَٱلَّذِينَ يَدْعُونَ مِن دُونِهِۦ لَا يَقْضُونَ بِشَىْءٍ ۗ إِنَّ ٱللَّهَ هُوَ ٱلسَّمِيعُ ٱلْبَصِيرُ</span>
<br><span class="ayah_translation">خداوند بحقّ داوری می‌کند، و معبودهایی را که غیر از او می‌خوانند هیچ گونه داوری ندارند؛ خداوند شنوا و بیناست!</span>

##### 21

<span class="ayah">۞ أَوَلَمْ يَسِيرُوا۟ فِى ٱلْأَرْضِ فَيَنظُرُوا۟ كَيْفَ كَانَ عَٰقِبَةُ ٱلَّذِينَ كَانُوا۟ مِن قَبْلِهِمْ ۚ كَانُوا۟ هُمْ أَشَدَّ مِنْهُمْ قُوَّةًۭ وَءَاثَارًۭا فِى ٱلْأَرْضِ فَأَخَذَهُمُ ٱللَّهُ بِذُنُوبِهِمْ وَمَا كَانَ لَهُم مِّنَ ٱللَّهِ مِن وَاقٍۢ</span>
<br><span class="ayah_translation">آیا آنها روی زمین سیر نکردند تا ببینند عاقبت کسانی که پیش از آنان بودند چگونه بود؟! آنها در قدرت و ایجاد آثار مهمّ در زمین از اینها برتر بودند؛ ولی خداوند ایشان را به گناهانشان گرفت، و در برابر عذاب او مدافعی نداشتند!</span>

##### 22

<span class="ayah">ذَٰلِكَ بِأَنَّهُمْ كَانَت تَّأْتِيهِمْ رُسُلُهُم بِٱلْبَيِّنَٰتِ فَكَفَرُوا۟ فَأَخَذَهُمُ ٱللَّهُ ۚ إِنَّهُۥ قَوِىٌّۭ شَدِيدُ ٱلْعِقَابِ</span>
<br><span class="ayah_translation">این برای آن بود که پیامبرانشان پیوسته با دلایل روشن به سراغشان می‌آمدند، ولی آنها انکار می‌کردند؛ خداوند هم آنان را گرفت (و کیفر داد) که او قوّی و مجازاتش شدید است!</span>

##### 23

<span class="ayah">وَلَقَدْ أَرْسَلْنَا مُوسَىٰ بِـَٔايَٰتِنَا وَسُلْطَٰنٍۢ مُّبِينٍ</span>
<br><span class="ayah_translation">ما موسی را با آیات خود و دلیل روشن فرستادیم...</span>

##### 24

<span class="ayah">إِلَىٰ فِرْعَوْنَ وَهَٰمَٰنَ وَقَٰرُونَ فَقَالُوا۟ سَٰحِرٌۭ كَذَّابٌۭ</span>
<br><span class="ayah_translation">بسوی فرعون و هامان و قارون؛ ولی آنها گفتند: «او ساحری بسیار دروغگو است!»</span>

##### 25

<span class="ayah">فَلَمَّا جَآءَهُم بِٱلْحَقِّ مِنْ عِندِنَا قَالُوا۟ ٱقْتُلُوٓا۟ أَبْنَآءَ ٱلَّذِينَ ءَامَنُوا۟ مَعَهُۥ وَٱسْتَحْيُوا۟ نِسَآءَهُمْ ۚ وَمَا كَيْدُ ٱلْكَٰفِرِينَ إِلَّا فِى ضَلَٰلٍۢ</span>
<br><span class="ayah_translation">و هنگامی که حقّ را از سوی ما برای آنها آورد، گفتند: «پسران کسانی را که با موسی ایمان آورده‌اند بکشید و زنانشان را (برای اسارت و خدمت) زنده بگذارید!» امّا نقشه کافران جز در گمراهی نیست (و نقش بر آب می‌شود).</span>

##### 26

<span class="ayah">وَقَالَ فِرْعَوْنُ ذَرُونِىٓ أَقْتُلْ مُوسَىٰ وَلْيَدْعُ رَبَّهُۥٓ ۖ إِنِّىٓ أَخَافُ أَن يُبَدِّلَ دِينَكُمْ أَوْ أَن يُظْهِرَ فِى ٱلْأَرْضِ ٱلْفَسَادَ</span>
<br><span class="ayah_translation">و فرعون گفت: «بگذارید موسی را بکشم، و او پروردگارش را بخواند (تا نجاتش دهد)! زیرا من می‌ترسم که آیین شما را دگرگون سازد، و یا در این سرزمین فساد بر پا کند!»</span>

##### 27

<span class="ayah">وَقَالَ مُوسَىٰٓ إِنِّى عُذْتُ بِرَبِّى وَرَبِّكُم مِّن كُلِّ مُتَكَبِّرٍۢ لَّا يُؤْمِنُ بِيَوْمِ ٱلْحِسَابِ</span>
<br><span class="ayah_translation">موسی گفت: «من به پروردگارم و پروردگار شما پناه می‌برم از هر متکبّری که به روز حساب ایمان نمی‌آورد!»</span>

##### 28

<span class="ayah">وَقَالَ رَجُلٌۭ مُّؤْمِنٌۭ مِّنْ ءَالِ فِرْعَوْنَ يَكْتُمُ إِيمَٰنَهُۥٓ أَتَقْتُلُونَ رَجُلًا أَن يَقُولَ رَبِّىَ ٱللَّهُ وَقَدْ جَآءَكُم بِٱلْبَيِّنَٰتِ مِن رَّبِّكُمْ ۖ وَإِن يَكُ كَٰذِبًۭا فَعَلَيْهِ كَذِبُهُۥ ۖ وَإِن يَكُ صَادِقًۭا يُصِبْكُم بَعْضُ ٱلَّذِى يَعِدُكُمْ ۖ إِنَّ ٱللَّهَ لَا يَهْدِى مَنْ هُوَ مُسْرِفٌۭ كَذَّابٌۭ</span>
<br><span class="ayah_translation">و مرد مؤمنی از آل فرعون که ایمان خود را پنهان می‌داشت گفت: «آیا می‌خواهید مردی را بکشید بخاطر اینکه می‌گوید: پروردگار من «اللّه» است، در حالی که دلایل روشنی از سوی پروردگارتان برای شما آورده است؟! اگر دروغگو باشد، دروغش دامن خودش را خواهد گرفت؛ و اگر راستگو باشد، (لااقل) بعضی از عذابهایی را که وعده می‌دهد به شما خواهد رسید؛ خداوند کسی را که اسرافکار و بسیار دروغگوست هدایت نمی‌کند.</span>

##### 29

<span class="ayah">يَٰقَوْمِ لَكُمُ ٱلْمُلْكُ ٱلْيَوْمَ ظَٰهِرِينَ فِى ٱلْأَرْضِ فَمَن يَنصُرُنَا مِنۢ بَأْسِ ٱللَّهِ إِن جَآءَنَا ۚ قَالَ فِرْعَوْنُ مَآ أُرِيكُمْ إِلَّا مَآ أَرَىٰ وَمَآ أَهْدِيكُمْ إِلَّا سَبِيلَ ٱلرَّشَادِ</span>
<br><span class="ayah_translation">ای قوم من! امروز حکومت از آن شماست و در این سرزمین پیروزید؛ اگر عذاب خدا به سراغ ما آید، چه کسی ما را یاری خواهد کرد؟!» فرعون گفت: «من جز آنچه را معتقدم به شما ارائه نمی‌دهم، و شما را جز به راه صحیح راهنمایی نمی‌کنم! (دستور، همان قتل موسی است!)»</span>

##### 30

<span class="ayah">وَقَالَ ٱلَّذِىٓ ءَامَنَ يَٰقَوْمِ إِنِّىٓ أَخَافُ عَلَيْكُم مِّثْلَ يَوْمِ ٱلْأَحْزَابِ</span>
<br><span class="ayah_translation">آن مرد باایمان گفت: «ای قوم من! من بر شما از روزی همانند روز (عذاب) اقوام پیشین بیمناکم!</span>

##### 31

<span class="ayah">مِثْلَ دَأْبِ قَوْمِ نُوحٍۢ وَعَادٍۢ وَثَمُودَ وَٱلَّذِينَ مِنۢ بَعْدِهِمْ ۚ وَمَا ٱللَّهُ يُرِيدُ ظُلْمًۭا لِّلْعِبَادِ</span>
<br><span class="ayah_translation">و از عادتی همچون عادت قوم نوح و عاد و ثمود و کسانی که بعد از آنان بودند (از شرک و کفر و طغیان) می‌ترسم؛ و خداوند ظلم و ستمی بر بندگانش نمی‌خواهد.</span>

##### 32

<span class="ayah">وَيَٰقَوْمِ إِنِّىٓ أَخَافُ عَلَيْكُمْ يَوْمَ ٱلتَّنَادِ</span>
<br><span class="ayah_translation">ای قوم من! من بر شما از روزی که مردم یکدیگر را صدا می‌زنند (و از هم یاری می‌طلبند و صدایشان به جایی نمی‌رسد) بیمناکم!</span>

##### 33

<span class="ayah">يَوْمَ تُوَلُّونَ مُدْبِرِينَ مَا لَكُم مِّنَ ٱللَّهِ مِنْ عَاصِمٍۢ ۗ وَمَن يُضْلِلِ ٱللَّهُ فَمَا لَهُۥ مِنْ هَادٍۢ</span>
<br><span class="ayah_translation">همان روزی که روی می‌گردانید و فرار می‌کنید؛ امّا هیچ پناهگاهی در برابر عذاب خداوند برای شما نیست؛ و هر کس را خداوند (بخاطر اعمالش) گمراه سازد، هدایت‌کننده‌ای برای او نیست!</span>

##### 34

<span class="ayah">وَلَقَدْ جَآءَكُمْ يُوسُفُ مِن قَبْلُ بِٱلْبَيِّنَٰتِ فَمَا زِلْتُمْ فِى شَكٍّۢ مِّمَّا جَآءَكُم بِهِۦ ۖ حَتَّىٰٓ إِذَا هَلَكَ قُلْتُمْ لَن يَبْعَثَ ٱللَّهُ مِنۢ بَعْدِهِۦ رَسُولًۭا ۚ كَذَٰلِكَ يُضِلُّ ٱللَّهُ مَنْ هُوَ مُسْرِفٌۭ مُّرْتَابٌ</span>
<br><span class="ayah_translation">پیش از این یوسف دلایل روشن برای شما آورد، ولی شما همچنان در آنچه او برای شما آورده بود تردید داشتید؛ تا زمانی که از دنیا رفت، گفتید: هرگز خداوند بعد از او پیامبری مبعوث نخواهد کرد! این گونه خداوند هر اسرافکار تردیدکننده‌ای را گمراه می‌سازد!</span>

##### 35

<span class="ayah">ٱلَّذِينَ يُجَٰدِلُونَ فِىٓ ءَايَٰتِ ٱللَّهِ بِغَيْرِ سُلْطَٰنٍ أَتَىٰهُمْ ۖ كَبُرَ مَقْتًا عِندَ ٱللَّهِ وَعِندَ ٱلَّذِينَ ءَامَنُوا۟ ۚ كَذَٰلِكَ يَطْبَعُ ٱللَّهُ عَلَىٰ كُلِّ قَلْبِ مُتَكَبِّرٍۢ جَبَّارٍۢ</span>
<br><span class="ayah_translation">همانها که در آیات خدا بی‌آنکه دلیلی برایشان آمده باشد به مجادله برمی‌خیزند؛ (این کارشان) خشم عظیمی نزد خداوند و نزد آنان که ایمان آورده‌اند به بار می‌آورد؛ این گونه خداوند بر دل هر متکبّر جبّاری مُهر می‌نهد!»</span>

##### 36

<span class="ayah">وَقَالَ فِرْعَوْنُ يَٰهَٰمَٰنُ ٱبْنِ لِى صَرْحًۭا لَّعَلِّىٓ أَبْلُغُ ٱلْأَسْبَٰبَ</span>
<br><span class="ayah_translation">فرعون گفت: «ای هامان! برای من بنای مرتفعی بساز، شاید به وسایلی دست یابم،</span>

##### 37

<span class="ayah">أَسْبَٰبَ ٱلسَّمَٰوَٰتِ فَأَطَّلِعَ إِلَىٰٓ إِلَٰهِ مُوسَىٰ وَإِنِّى لَأَظُنُّهُۥ كَٰذِبًۭا ۚ وَكَذَٰلِكَ زُيِّنَ لِفِرْعَوْنَ سُوٓءُ عَمَلِهِۦ وَصُدَّ عَنِ ٱلسَّبِيلِ ۚ وَمَا كَيْدُ فِرْعَوْنَ إِلَّا فِى تَبَابٍۢ</span>
<br><span class="ayah_translation">وسایل (صعود به) آسمانها تا از خدای موسی آگاه شوم؛ هر چند گمان می‌کنم او دروغگو باشد!» اینچنین اعمال بد فرعون در نظرش آراسته جلوه کرد و از راه حق باز داشته شد؛ و توطئه فرعون (و همفکران او) جز به نابودی نمی‌انجامد!</span>

##### 38

<span class="ayah">وَقَالَ ٱلَّذِىٓ ءَامَنَ يَٰقَوْمِ ٱتَّبِعُونِ أَهْدِكُمْ سَبِيلَ ٱلرَّشَادِ</span>
<br><span class="ayah_translation">کسی که (از قوم فرعون) ایمان آورده بود گفت: «ای قوم من! از من پیروی کنید تا شما را به راه درست هدایت کنم.</span>

##### 39

<span class="ayah">يَٰقَوْمِ إِنَّمَا هَٰذِهِ ٱلْحَيَوٰةُ ٱلدُّنْيَا مَتَٰعٌۭ وَإِنَّ ٱلْءَاخِرَةَ هِىَ دَارُ ٱلْقَرَارِ</span>
<br><span class="ayah_translation">ای قوم من! این زندگی دنیا، تنها متاع زودگذری است؛ و آخرت سرای همیشگی است!</span>

##### 40

<span class="ayah">مَنْ عَمِلَ سَيِّئَةًۭ فَلَا يُجْزَىٰٓ إِلَّا مِثْلَهَا ۖ وَمَنْ عَمِلَ صَٰلِحًۭا مِّن ذَكَرٍ أَوْ أُنثَىٰ وَهُوَ مُؤْمِنٌۭ فَأُو۟لَٰٓئِكَ يَدْخُلُونَ ٱلْجَنَّةَ يُرْزَقُونَ فِيهَا بِغَيْرِ حِسَابٍۢ</span>
<br><span class="ayah_translation">هر کس بدی کند، جز بمانند آن کیفر داده نمی‌شود؛ ولی هر کس کار شایسته‌ای انجام دهد -خواه مرد یا زن- در حالی که مؤمن باشد آنها وارد بهشت می‌شوند و در آن روزی بی‌حسابی به آنها داده خواهد شد.</span>

##### 41

<span class="ayah">۞ وَيَٰقَوْمِ مَا لِىٓ أَدْعُوكُمْ إِلَى ٱلنَّجَوٰةِ وَتَدْعُونَنِىٓ إِلَى ٱلنَّارِ</span>
<br><span class="ayah_translation">ای قوم من! چرا من شما را به سوی نجات دعوت می‌کنم، امّا شما مرا بسوی آتش فرا می‌خوانید؟!</span>

##### 42

<span class="ayah">تَدْعُونَنِى لِأَكْفُرَ بِٱللَّهِ وَأُشْرِكَ بِهِۦ مَا لَيْسَ لِى بِهِۦ عِلْمٌۭ وَأَنَا۠ أَدْعُوكُمْ إِلَى ٱلْعَزِيزِ ٱلْغَفَّٰرِ</span>
<br><span class="ayah_translation">مرا دعوت می‌کنید که به خداوند یگانه کافر شوم و همتایی که به آن علم ندارم برای او قرار دهم، در حالی که من شما را بسوی خداوند عزیز غفّار دعوت می‌کنم!</span>

##### 43

<span class="ayah">لَا جَرَمَ أَنَّمَا تَدْعُونَنِىٓ إِلَيْهِ لَيْسَ لَهُۥ دَعْوَةٌۭ فِى ٱلدُّنْيَا وَلَا فِى ٱلْءَاخِرَةِ وَأَنَّ مَرَدَّنَآ إِلَى ٱللَّهِ وَأَنَّ ٱلْمُسْرِفِينَ هُمْ أَصْحَٰبُ ٱلنَّارِ</span>
<br><span class="ayah_translation">قطعاً آنچه مرا بسوی آن می‌خوانید، نه دعوت (و حاکمیّتی) در دنیا دارد و نه در آخرت؛ و تنها بازگشت ما در قیامت بسوی خداست؛ و مسرفان اهل آتشند!</span>

##### 44

<span class="ayah">فَسَتَذْكُرُونَ مَآ أَقُولُ لَكُمْ ۚ وَأُفَوِّضُ أَمْرِىٓ إِلَى ٱللَّهِ ۚ إِنَّ ٱللَّهَ بَصِيرٌۢ بِٱلْعِبَادِ</span>
<br><span class="ayah_translation">و بزودی آنچه را به شما می‌گویم به خاطر خواهید آورد! من کار خود را به خدا واگذارم که خداوند نسبت به بندگانش بیناست!»</span>

##### 45

<span class="ayah">فَوَقَىٰهُ ٱللَّهُ سَيِّـَٔاتِ مَا مَكَرُوا۟ ۖ وَحَاقَ بِـَٔالِ فِرْعَوْنَ سُوٓءُ ٱلْعَذَابِ</span>
<br><span class="ayah_translation">خداوند او را از نقشه‌های سوء آنها نگه داشت، و عذاب شدید بر آل فرعون وارد شد!</span>

##### 46

<span class="ayah">ٱلنَّارُ يُعْرَضُونَ عَلَيْهَا غُدُوًّۭا وَعَشِيًّۭا ۖ وَيَوْمَ تَقُومُ ٱلسَّاعَةُ أَدْخِلُوٓا۟ ءَالَ فِرْعَوْنَ أَشَدَّ ٱلْعَذَابِ</span>
<br><span class="ayah_translation">عذاب آنها آتش است که هر صبح و شام بر آن عرضه می‌شوند؛ و روزی که قیامت برپا شود (می‌فرماید:) «آل فرعون را در سخت‌ترین عذابها وارد کنید!»</span>

##### 47

<span class="ayah">وَإِذْ يَتَحَآجُّونَ فِى ٱلنَّارِ فَيَقُولُ ٱلضُّعَفَٰٓؤُا۟ لِلَّذِينَ ٱسْتَكْبَرُوٓا۟ إِنَّا كُنَّا لَكُمْ تَبَعًۭا فَهَلْ أَنتُم مُّغْنُونَ عَنَّا نَصِيبًۭا مِّنَ ٱلنَّارِ</span>
<br><span class="ayah_translation">به خاطر بیاور هنگامی را که در آتش دوزخ با هم محاجّه می‌کنند؛ ضعیفان به مستکبران می‌گویند: «ما پیرو شما بودیم، آیا شما (امروز) سهمی از آتش را بجای ما پذیرا می‌شوید؟!»</span>

##### 48

<span class="ayah">قَالَ ٱلَّذِينَ ٱسْتَكْبَرُوٓا۟ إِنَّا كُلٌّۭ فِيهَآ إِنَّ ٱللَّهَ قَدْ حَكَمَ بَيْنَ ٱلْعِبَادِ</span>
<br><span class="ayah_translation">مستکبران می‌گویند: «ما همگی در آن هستیم، زیرا خداوند در میان بندگانش (بعدالت) حکم کرده است!»</span>

##### 49

<span class="ayah">وَقَالَ ٱلَّذِينَ فِى ٱلنَّارِ لِخَزَنَةِ جَهَنَّمَ ٱدْعُوا۟ رَبَّكُمْ يُخَفِّفْ عَنَّا يَوْمًۭا مِّنَ ٱلْعَذَابِ</span>
<br><span class="ayah_translation">و آنها که در آتشند به مأموران دوزخ می‌گویند: «از پروردگارتان بخواهید یک روز عذاب را از ما بردارد!»</span>

##### 50

<span class="ayah">قَالُوٓا۟ أَوَلَمْ تَكُ تَأْتِيكُمْ رُسُلُكُم بِٱلْبَيِّنَٰتِ ۖ قَالُوا۟ بَلَىٰ ۚ قَالُوا۟ فَٱدْعُوا۟ ۗ وَمَا دُعَٰٓؤُا۟ ٱلْكَٰفِرِينَ إِلَّا فِى ضَلَٰلٍ</span>
<br><span class="ayah_translation">آنها می‌گویند: «آیا پیامبران شما دلایل روشن برایتان نیاوردند؟!» می‌گویند: «آری!» آنها می‌گویند: «پس هر چه می‌خواهید (خدا را) بخوانید؛ ولی دعای کافران (به جایی نمی‌رسد و) جز در ضلالت نیست!»</span>

##### 51

<span class="ayah">إِنَّا لَنَنصُرُ رُسُلَنَا وَٱلَّذِينَ ءَامَنُوا۟ فِى ٱلْحَيَوٰةِ ٱلدُّنْيَا وَيَوْمَ يَقُومُ ٱلْأَشْهَٰدُ</span>
<br><span class="ayah_translation">ما به یقین پیامبران خود و کسانی را که ایمان آورده‌اند، در زندگی دنیا و (در آخرت) روزی که گواهان به پا می‌خیزند یاری می‌دهیم!</span>

##### 52

<span class="ayah">يَوْمَ لَا يَنفَعُ ٱلظَّٰلِمِينَ مَعْذِرَتُهُمْ ۖ وَلَهُمُ ٱللَّعْنَةُ وَلَهُمْ سُوٓءُ ٱلدَّارِ</span>
<br><span class="ayah_translation">روزی که عذرخواهی ظالمان سودی به حالشان نمی‌بخشد؛ و لعنت خدا برای آنها، و خانه (و جایگاه) بد نیز برای آنان است.</span>

##### 53

<span class="ayah">وَلَقَدْ ءَاتَيْنَا مُوسَى ٱلْهُدَىٰ وَأَوْرَثْنَا بَنِىٓ إِسْرَٰٓءِيلَ ٱلْكِتَٰبَ</span>
<br><span class="ayah_translation">و ما به موسی هدایت بخشیدیم، و بنی اسرائیل را وارثان کتاب (تورات) قرار دادیم.</span>

##### 54

<span class="ayah">هُدًۭى وَذِكْرَىٰ لِأُو۟لِى ٱلْأَلْبَٰبِ</span>
<br><span class="ayah_translation">کتابی که مایه هدایت و تذکّر برای صاحبان عقل بود!</span>

##### 55

<span class="ayah">فَٱصْبِرْ إِنَّ وَعْدَ ٱللَّهِ حَقٌّۭ وَٱسْتَغْفِرْ لِذَنۢبِكَ وَسَبِّحْ بِحَمْدِ رَبِّكَ بِٱلْعَشِىِّ وَٱلْإِبْكَٰرِ</span>
<br><span class="ayah_translation">پس (ای پیامبر!) صبر و شکیبایی پیشه کن که وعده خدا حقّ است، و برای گناهت استغفار کن، و هر صبح و شام تسبیح و حمد پروردگارت را بجا آور!</span>

##### 56

<span class="ayah">إِنَّ ٱلَّذِينَ يُجَٰدِلُونَ فِىٓ ءَايَٰتِ ٱللَّهِ بِغَيْرِ سُلْطَٰنٍ أَتَىٰهُمْ ۙ إِن فِى صُدُورِهِمْ إِلَّا كِبْرٌۭ مَّا هُم بِبَٰلِغِيهِ ۚ فَٱسْتَعِذْ بِٱللَّهِ ۖ إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْبَصِيرُ</span>
<br><span class="ayah_translation">کسانی که در آیات خداوند بدون دلیلی که برای آنها آمده باشد ستیزه‌جویی می‌کنند، در سینه‌هایشان فقط تکبّر (و غرور) است، و هرگز به خواسته خود نخواهند رسید، پس به خدا پناه بر که او شنوا و بیناست!</span>

##### 57

<span class="ayah">لَخَلْقُ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ أَكْبَرُ مِنْ خَلْقِ ٱلنَّاسِ وَلَٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ</span>
<br><span class="ayah_translation">آفرینش آسمانها و زمین از آفرینش انسانها مهمتر است، ولی بیشتر مردم نمی‌دانند.</span>

##### 58

<span class="ayah">وَمَا يَسْتَوِى ٱلْأَعْمَىٰ وَٱلْبَصِيرُ وَٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ وَلَا ٱلْمُسِىٓءُ ۚ قَلِيلًۭا مَّا تَتَذَكَّرُونَ</span>
<br><span class="ayah_translation">هرگز نابینا و بینا یکسان نیستند؛ همچنین کسانی که ایمان آورده، و اعمال صالح انجام داده‌اند با بدکاران یکسان نخواهند بود؛ امّا کمتر متذکّر می‌شوید!</span>

##### 59

<span class="ayah">إِنَّ ٱلسَّاعَةَ لَءَاتِيَةٌۭ لَّا رَيْبَ فِيهَا وَلَٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يُؤْمِنُونَ</span>
<br><span class="ayah_translation">روز قیامت به یقین آمدنی است، و شکّی در آن نیست؛ ولی اکثر مردم ایمان نمی آورند!</span>

##### 60

<span class="ayah">وَقَالَ رَبُّكُمُ ٱدْعُونِىٓ أَسْتَجِبْ لَكُمْ ۚ إِنَّ ٱلَّذِينَ يَسْتَكْبِرُونَ عَنْ عِبَادَتِى سَيَدْخُلُونَ جَهَنَّمَ دَاخِرِينَ</span>
<br><span class="ayah_translation">پروردگار شما گفته است: «مرا بخوانید تا (دعای) شما را بپذیرم! کسانی که از عبادت من تکبّر می‌ورزند به زودی با ذلّت وارد دوزخ می‌شوند!»</span>

##### 61

<span class="ayah">ٱللَّهُ ٱلَّذِى جَعَلَ لَكُمُ ٱلَّيْلَ لِتَسْكُنُوا۟ فِيهِ وَٱلنَّهَارَ مُبْصِرًا ۚ إِنَّ ٱللَّهَ لَذُو فَضْلٍ عَلَى ٱلنَّاسِ وَلَٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَشْكُرُونَ</span>
<br><span class="ayah_translation">خداوند کسی است که شب را برای شما آفرید تا در آن بیاسایید، و روز را روشنی‌بخش قرار داد؛ خداوند نسبت به مردم صاحب فضل و کرم است؛ ولی بیشتر مردم شکرگزاری نمی‌کنند!</span>

##### 62

<span class="ayah">ذَٰلِكُمُ ٱللَّهُ رَبُّكُمْ خَٰلِقُ كُلِّ شَىْءٍۢ لَّآ إِلَٰهَ إِلَّا هُوَ ۖ فَأَنَّىٰ تُؤْفَكُونَ</span>
<br><span class="ayah_translation">این است خداوند، پروردگار شما که آفریننده همه چیز است؛ هیچ معبودی جز او نیست؛ با این حال چگونه از راه حق منحرف می‌شوید؟!</span>

##### 63

<span class="ayah">كَذَٰلِكَ يُؤْفَكُ ٱلَّذِينَ كَانُوا۟ بِـَٔايَٰتِ ٱللَّهِ يَجْحَدُونَ</span>
<br><span class="ayah_translation">اینچنین کسانی که آیات خدا را انکار می‌کردند (از راه راست) منحرف می‌شوند!</span>

##### 64

<span class="ayah">ٱللَّهُ ٱلَّذِى جَعَلَ لَكُمُ ٱلْأَرْضَ قَرَارًۭا وَٱلسَّمَآءَ بِنَآءًۭ وَصَوَّرَكُمْ فَأَحْسَنَ صُوَرَكُمْ وَرَزَقَكُم مِّنَ ٱلطَّيِّبَٰتِ ۚ ذَٰلِكُمُ ٱللَّهُ رَبُّكُمْ ۖ فَتَبَارَكَ ٱللَّهُ رَبُّ ٱلْعَٰلَمِينَ</span>
<br><span class="ayah_translation">خداوند کسی است که زمین را برای شما جایگاه امن و آرامش قرار داد و آسمان را همچون سقفی (بالای سرتان)؛ و شما را صورتگری کرد، و صورتتان را نیکو آفرید؛ و از چیزهایی پاکیزه به شما روزی داد؛ این است خداوند پروردگار شما! جاوید و پربرکت است خداوندی که پروردگار عالمیان است!</span>

##### 65

<span class="ayah">هُوَ ٱلْحَىُّ لَآ إِلَٰهَ إِلَّا هُوَ فَٱدْعُوهُ مُخْلِصِينَ لَهُ ٱلدِّينَ ۗ ٱلْحَمْدُ لِلَّهِ رَبِّ ٱلْعَٰلَمِينَ</span>
<br><span class="ayah_translation">زنده (واقعی) اوست؛ معبودی جز او نیست؛ پس او را بخوانید در حالی که دین خود را برای او خالص کرده‌اید! ستایش مخصوص خداوندی است که پروردگار جهانیان است!</span>

##### 66

<span class="ayah">۞ قُلْ إِنِّى نُهِيتُ أَنْ أَعْبُدَ ٱلَّذِينَ تَدْعُونَ مِن دُونِ ٱللَّهِ لَمَّا جَآءَنِىَ ٱلْبَيِّنَٰتُ مِن رَّبِّى وَأُمِرْتُ أَنْ أُسْلِمَ لِرَبِّ ٱلْعَٰلَمِينَ</span>
<br><span class="ayah_translation">بگو: «من نهی شده‌ام از اینکه معبودهایی را که شما غیر از خدا می‌خوانید بپرستم، چون دلایل روشن از جانب پروردگارم برای من آمده است؛ و مأمورم که تنها در برابر پروردگار عالمیان تسلیم باشم!»</span>

##### 67

<span class="ayah">هُوَ ٱلَّذِى خَلَقَكُم مِّن تُرَابٍۢ ثُمَّ مِن نُّطْفَةٍۢ ثُمَّ مِنْ عَلَقَةٍۢ ثُمَّ يُخْرِجُكُمْ طِفْلًۭا ثُمَّ لِتَبْلُغُوٓا۟ أَشُدَّكُمْ ثُمَّ لِتَكُونُوا۟ شُيُوخًۭا ۚ وَمِنكُم مَّن يُتَوَفَّىٰ مِن قَبْلُ ۖ وَلِتَبْلُغُوٓا۟ أَجَلًۭا مُّسَمًّۭى وَلَعَلَّكُمْ تَعْقِلُونَ</span>
<br><span class="ayah_translation">او کسی است که شما را از خاک آفرید، سپس از نطفه، سپس از علقه (خون بسته شده)، سپس شما را بصورت طفلی (از شکم مادر) بیرون می‌فرستد، بعد به مرحله کمال قوّت خود می‌رسید، و بعد از آن پیر می‌شوید و (در این میان) گروهی از شما پیش از رسیدن به این مرحله می‌میرند و در نهایت به سرآمد عمر خود می‌رسید؛ و شاید تعقّل کنید!</span>

##### 68

<span class="ayah">هُوَ ٱلَّذِى يُحْىِۦ وَيُمِيتُ ۖ فَإِذَا قَضَىٰٓ أَمْرًۭا فَإِنَّمَا يَقُولُ لَهُۥ كُن فَيَكُونُ</span>
<br><span class="ayah_translation">او کسی است که زنده می‌کند و می‌میراند؛ و هنگامی که کاری را مقرّر کند، تنها به آن می‌گوید: «موجود باش!» بی‌درنگ موجود می‌شود!</span>

##### 69

<span class="ayah">أَلَمْ تَرَ إِلَى ٱلَّذِينَ يُجَٰدِلُونَ فِىٓ ءَايَٰتِ ٱللَّهِ أَنَّىٰ يُصْرَفُونَ</span>
<br><span class="ayah_translation">آیا ندیدی کسانی را که در آیات خدا مجادله می‌کنند، چگونه از راه حقّ منحرف می شوند؟!</span>

##### 70

<span class="ayah">ٱلَّذِينَ كَذَّبُوا۟ بِٱلْكِتَٰبِ وَبِمَآ أَرْسَلْنَا بِهِۦ رُسُلَنَا ۖ فَسَوْفَ يَعْلَمُونَ</span>
<br><span class="ayah_translation">همان کسانی که کتاب (آسمانی) و آنچه رسولان خود را بدان فرستاده‌ایم تکذیب کردند؛ امّا بزودی (نتیجه کار خود را) می‌دانند!</span>

##### 71

<span class="ayah">إِذِ ٱلْأَغْلَٰلُ فِىٓ أَعْنَٰقِهِمْ وَٱلسَّلَٰسِلُ يُسْحَبُونَ</span>
<br><span class="ayah_translation">در آن هنگام که غل و زنجیرها بر گردن آنان قرار گرفته و آنها را می‌کشند...</span>

##### 72

<span class="ayah">فِى ٱلْحَمِيمِ ثُمَّ فِى ٱلنَّارِ يُسْجَرُونَ</span>
<br><span class="ayah_translation">و در آب جوشان وارد می‌کنند؛ سپس در آتش دوزخ افروخته می‌شوند!</span>

##### 73

<span class="ayah">ثُمَّ قِيلَ لَهُمْ أَيْنَ مَا كُنتُمْ تُشْرِكُونَ</span>
<br><span class="ayah_translation">سپس به آنها گفته می‌شود: «کجایند آنچه را همتای خدا قرار می‌دادید،</span>

##### 74

<span class="ayah">مِن دُونِ ٱللَّهِ ۖ قَالُوا۟ ضَلُّوا۟ عَنَّا بَل لَّمْ نَكُن نَّدْعُوا۟ مِن قَبْلُ شَيْـًۭٔا ۚ كَذَٰلِكَ يُضِلُّ ٱللَّهُ ٱلْكَٰفِرِينَ</span>
<br><span class="ayah_translation">همان معبودهایی را که جز خدا پرستش می‌کردید؟!» آنها می‌گویند: «همه از نظر ما پنهان و گم شدند؛ بلکه ما اصلاً پیش از این چیزی را پرستش نمی‌کردیم»! این گونه خداوند کافران را گمراه می‌سازد!</span>

##### 75

<span class="ayah">ذَٰلِكُم بِمَا كُنتُمْ تَفْرَحُونَ فِى ٱلْأَرْضِ بِغَيْرِ ٱلْحَقِّ وَبِمَا كُنتُمْ تَمْرَحُونَ</span>
<br><span class="ayah_translation">این (عذاب) بخاطر آن است که بناحقّ در زمین شادی می‌کردید و از روی غرور و مستی به خوشحالی می‌پرداختید!</span>

##### 76

<span class="ayah">ٱدْخُلُوٓا۟ أَبْوَٰبَ جَهَنَّمَ خَٰلِدِينَ فِيهَا ۖ فَبِئْسَ مَثْوَى ٱلْمُتَكَبِّرِينَ</span>
<br><span class="ayah_translation">از درهای جهنّم وارد شوید و جاودانه در آن بمانید؛ و چه بد است جایگاه متکبّران!</span>

##### 77

<span class="ayah">فَٱصْبِرْ إِنَّ وَعْدَ ٱللَّهِ حَقٌّۭ ۚ فَإِمَّا نُرِيَنَّكَ بَعْضَ ٱلَّذِى نَعِدُهُمْ أَوْ نَتَوَفَّيَنَّكَ فَإِلَيْنَا يُرْجَعُونَ</span>
<br><span class="ayah_translation">پس (ای پیامبر) صبر کن که وعده خدا حقّ است؛ و هرگاه قسمتی از مجازاتهایی را که به آنها وعده داده‌ایم در حال حیاتت به تو ارائه دهیم، یا تو را (پیش از آن) از دنیا ببریم (مهمّ نیست)؛ چرا که همه آنان را تنها بسوی ما باز می‌گردانند!</span>

##### 78

<span class="ayah">وَلَقَدْ أَرْسَلْنَا رُسُلًۭا مِّن قَبْلِكَ مِنْهُم مَّن قَصَصْنَا عَلَيْكَ وَمِنْهُم مَّن لَّمْ نَقْصُصْ عَلَيْكَ ۗ وَمَا كَانَ لِرَسُولٍ أَن يَأْتِىَ بِـَٔايَةٍ إِلَّا بِإِذْنِ ٱللَّهِ ۚ فَإِذَا جَآءَ أَمْرُ ٱللَّهِ قُضِىَ بِٱلْحَقِّ وَخَسِرَ هُنَالِكَ ٱلْمُبْطِلُونَ</span>
<br><span class="ayah_translation">ما پیش از تو رسولانی فرستادیم؛ سرگذشت گروهی از آنان را برای تو بازگفته، و گروهی را برای تو بازگو نکرده‌ایم؛ و هیچ پیامبری حق نداشت معجزه‌ای جز بفرمان خدا بیاورد و هنگامی که فرمان خداوند (برای مجازات آنها) صادر شود، بحق داوری خواهد شد؛ و آنجا اهل باطل زیان خواهند کرد!</span>

##### 79

<span class="ayah">ٱللَّهُ ٱلَّذِى جَعَلَ لَكُمُ ٱلْأَنْعَٰمَ لِتَرْكَبُوا۟ مِنْهَا وَمِنْهَا تَأْكُلُونَ</span>
<br><span class="ayah_translation">خداوند کسی است که چهارپایان را برای شما آفرید تا بعضی را سوار شوید و از بعضی تغذیه کنید.</span>

##### 80

<span class="ayah">وَلَكُمْ فِيهَا مَنَٰفِعُ وَلِتَبْلُغُوا۟ عَلَيْهَا حَاجَةًۭ فِى صُدُورِكُمْ وَعَلَيْهَا وَعَلَى ٱلْفُلْكِ تُحْمَلُونَ</span>
<br><span class="ayah_translation">و برای شما در آنها منافع بسیاری (جز اینها) است، تا بوسیله آنها به مقصدی که در دل دارید برسید؛ و بر آنها و بر کشتیها سوار می‌شوید.</span>

##### 81

<span class="ayah">وَيُرِيكُمْ ءَايَٰتِهِۦ فَأَىَّ ءَايَٰتِ ٱللَّهِ تُنكِرُونَ</span>
<br><span class="ayah_translation">او آیاتش را همواره به شما نشان می‌دهد؛ پس کدام یک از آیات او را انکار می‌کنید؟!</span>

##### 82

<span class="ayah">أَفَلَمْ يَسِيرُوا۟ فِى ٱلْأَرْضِ فَيَنظُرُوا۟ كَيْفَ كَانَ عَٰقِبَةُ ٱلَّذِينَ مِن قَبْلِهِمْ ۚ كَانُوٓا۟ أَكْثَرَ مِنْهُمْ وَأَشَدَّ قُوَّةًۭ وَءَاثَارًۭا فِى ٱلْأَرْضِ فَمَآ أَغْنَىٰ عَنْهُم مَّا كَانُوا۟ يَكْسِبُونَ</span>
<br><span class="ayah_translation">آیا روی زمین سیر نکردند تا ببینند عاقبت کسانی که پیش از آنها بودند چه شد؟! همانها که نفراتشان از اینها بیشتر، و نیرو و آثارشان در زمین فزونتر بود؛ امّا هرگز آنچه را به دست می‌آوردند نتوانست آنها را بی‌نیاز سازد (و عذاب الهی را از آنان دور کند)!</span>

##### 83

<span class="ayah">فَلَمَّا جَآءَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَٰتِ فَرِحُوا۟ بِمَا عِندَهُم مِّنَ ٱلْعِلْمِ وَحَاقَ بِهِم مَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ</span>
<br><span class="ayah_translation">هنگامی که رسولانشان دلایل روشنی برای آنان آوردند، به دانشی که خود داشتند خوشحال بودند (و غیر آن را هیچ می‌شمردند)؛ ولی آنچه را (از عذاب) به تمسخر می‌گرفتند آنان را فراگرفت!</span>

##### 84

<span class="ayah">فَلَمَّا رَأَوْا۟ بَأْسَنَا قَالُوٓا۟ ءَامَنَّا بِٱللَّهِ وَحْدَهُۥ وَكَفَرْنَا بِمَا كُنَّا بِهِۦ مُشْرِكِينَ</span>
<br><span class="ayah_translation">هنگامی که عذاب (شدید) ما را دیدند گفتند: «هم اکنون به خداوند یگانه ایمان آوردیم و به معبودهایی که همتای او می‌شمردیم کافر شدیم!»</span>

##### 85

<span class="ayah">فَلَمْ يَكُ يَنفَعُهُمْ إِيمَٰنُهُمْ لَمَّا رَأَوْا۟ بَأْسَنَا ۖ سُنَّتَ ٱللَّهِ ٱلَّتِى قَدْ خَلَتْ فِى عِبَادِهِۦ ۖ وَخَسِرَ هُنَالِكَ ٱلْكَٰفِرُونَ</span>
<br><span class="ayah_translation">امّا هنگامی که عذاب ما را مشاهده کردند، ایمانشان برای آنها سودی نداشت! این سنّت خداوند است که همواره در میان بندگانش اجرا شده، و آنجا کافران زیانکار شدند!</span>
