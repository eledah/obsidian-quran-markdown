▶ [[سورة الرعد]] | [[سورة الحجر]] ◀

<br>

##### 1

<span class="ayah">بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ الٓر ۚ كِتَٰبٌ أَنزَلْنَٰهُ إِلَيْكَ لِتُخْرِجَ ٱلنَّاسَ مِنَ ٱلظُّلُمَٰتِ إِلَى ٱلنُّورِ بِإِذْنِ رَبِّهِمْ إِلَىٰ صِرَٰطِ ٱلْعَزِيزِ ٱلْحَمِيدِ</span>
<br><span class="ayah_translation">الر، (این) کتابی است که بر تو نازل کردیم، تا مردم را از تاریکیها (ی شرک و ظلم و جهل،) به سوی روشنایی (ایمان و عدل و آگاهی،) بفرمان پروردگارشان در آوری، بسوی راه خداوند عزیز و حمید.</span>

##### 2

<span class="ayah">ٱللَّهِ ٱلَّذِى لَهُۥ مَا فِى ٱلسَّمَٰوَٰتِ وَمَا فِى ٱلْأَرْضِ ۗ وَوَيْلٌۭ لِّلْكَٰفِرِينَ مِنْ عَذَابٍۢ شَدِيدٍ</span>
<br><span class="ayah_translation">همان خدایی که آنچه در آسمانها و آنچه در زمین است، از آن اوست؛ وای بر کافران از مجازات شدید (الهی)!</span>

##### 3

<span class="ayah">ٱلَّذِينَ يَسْتَحِبُّونَ ٱلْحَيَوٰةَ ٱلدُّنْيَا عَلَى ٱلْءَاخِرَةِ وَيَصُدُّونَ عَن سَبِيلِ ٱللَّهِ وَيَبْغُونَهَا عِوَجًا ۚ أُو۟لَٰٓئِكَ فِى ضَلَٰلٍۭ بَعِيدٍۢ</span>
<br><span class="ayah_translation">همانها که زندگی دنیا را بر آخرت ترجیح می‌دهند؛ و (مردم را) از راه خدا باز می‌دارند؛ و می‌خواهند راه حق را منحرف سازند؛ آنها در گمراهی دوری هستند!</span>

##### 4

<span class="ayah">وَمَآ أَرْسَلْنَا مِن رَّسُولٍ إِلَّا بِلِسَانِ قَوْمِهِۦ لِيُبَيِّنَ لَهُمْ ۖ فَيُضِلُّ ٱللَّهُ مَن يَشَآءُ وَيَهْدِى مَن يَشَآءُ ۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ</span>
<br><span class="ayah_translation">ما هیچ پیامبری را، جز به زبان قومش، نفرستادیم؛ تا (حقایق را) برای آنها آشکار سازد؛ سپس خدا هر کس را بخواهد (و مستحق بداند) گمراه، و هر کس را بخواهد (و شایسته بداند) هدایت می‌کند؛ و او توانا و حکیم است.</span>

##### 5

<span class="ayah">وَلَقَدْ أَرْسَلْنَا مُوسَىٰ بِـَٔايَٰتِنَآ أَنْ أَخْرِجْ قَوْمَكَ مِنَ ٱلظُّلُمَٰتِ إِلَى ٱلنُّورِ وَذَكِّرْهُم بِأَيَّىٰمِ ٱللَّهِ ۚ إِنَّ فِى ذَٰلِكَ لَءَايَٰتٍۢ لِّكُلِّ صَبَّارٍۢ شَكُورٍۢ</span>
<br><span class="ayah_translation">ما موسی را با آیات خود فرستادیم؛ (و دستور دادیم:) قومت را از ظلمات به نور بیرون آر! و «ایّام اللّه» را به آنان یاد آور! در این، نشانه‌هایی است برای هر صبر کننده شکرگزار!</span>

##### 6

<span class="ayah">وَإِذْ قَالَ مُوسَىٰ لِقَوْمِهِ ٱذْكُرُوا۟ نِعْمَةَ ٱللَّهِ عَلَيْكُمْ إِذْ أَنجَىٰكُم مِّنْ ءَالِ فِرْعَوْنَ يَسُومُونَكُمْ سُوٓءَ ٱلْعَذَابِ وَيُذَبِّحُونَ أَبْنَآءَكُمْ وَيَسْتَحْيُونَ نِسَآءَكُمْ ۚ وَفِى ذَٰلِكُم بَلَآءٌۭ مِّن رَّبِّكُمْ عَظِيمٌۭ</span>
<br><span class="ayah_translation">و (به خاطر بیاور) هنگامی را که موسی به قومش گفت: «نعمت خدا را بر خود به یاد داشته باشید، زمانی که شما را از (چنگال) آل فرعون رهایی بخشید! همانها که شما را به بدترین وجهی عذاب می‌کردند؛ پسرانتان را سر می‌بریدند، و زنانتان را (برای خدمتگاری) زنده می‌گذاشتند؛ و در این، آزمایش بزرگی از طرف پروردگارتان برای شما بود!»</span>

##### 7

<span class="ayah">وَإِذْ تَأَذَّنَ رَبُّكُمْ لَئِن شَكَرْتُمْ لَأَزِيدَنَّكُمْ ۖ وَلَئِن كَفَرْتُمْ إِنَّ عَذَابِى لَشَدِيدٌۭ</span>
<br><span class="ayah_translation">و (همچنین به خاطر بیاورید) هنگامی را که پروردگارتان اعلام داشت: «اگر شکرگزاری کنید، (نعمت خود را) بر شما خواهم افزود؛ و اگر ناسپاسی کنید، مجازاتم شدید است!»</span>

##### 8

<span class="ayah">وَقَالَ مُوسَىٰٓ إِن تَكْفُرُوٓا۟ أَنتُمْ وَمَن فِى ٱلْأَرْضِ جَمِيعًۭا فَإِنَّ ٱللَّهَ لَغَنِىٌّ حَمِيدٌ</span>
<br><span class="ayah_translation">و موسی (به بنی اسرائیل) گفت: «اگر شما و همه مردم روی زمین کافر شوید، (به خدا زیانی نمی‌رسد؛ چرا که) خداوند، بی‌نیاز و شایسته ستایش است!»</span>

##### 9

<span class="ayah">أَلَمْ يَأْتِكُمْ نَبَؤُا۟ ٱلَّذِينَ مِن قَبْلِكُمْ قَوْمِ نُوحٍۢ وَعَادٍۢ وَثَمُودَ ۛ وَٱلَّذِينَ مِنۢ بَعْدِهِمْ ۛ لَا يَعْلَمُهُمْ إِلَّا ٱللَّهُ ۚ جَآءَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَٰتِ فَرَدُّوٓا۟ أَيْدِيَهُمْ فِىٓ أَفْوَٰهِهِمْ وَقَالُوٓا۟ إِنَّا كَفَرْنَا بِمَآ أُرْسِلْتُم بِهِۦ وَإِنَّا لَفِى شَكٍّۢ مِّمَّا تَدْعُونَنَآ إِلَيْهِ مُرِيبٍۢ</span>
<br><span class="ayah_translation">آیا خبر کسانی که پیش از شما بودند، به شما نرسید؟! «قوم نوح» و «عاد» و «ثمود» و آنها که پس از ایشان بودند؛ همانها که جز خداوند از آنان آگاه نیست؛ پیامبرانشان دلایل روشن برای آنان آوردند، ولی آنها (از روی تعجّب و استهزا) دست بر دهان گرفتند و گفتند: «ما به آنچه شما به آن فرستاده شده‌اید، کافریم! و نسبت به آنچه ما را به سوی آن می‌خوانید، شکّ و تردید داریم!»</span>

##### 10

<span class="ayah">۞ قَالَتْ رُسُلُهُمْ أَفِى ٱللَّهِ شَكٌّۭ فَاطِرِ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۖ يَدْعُوكُمْ لِيَغْفِرَ لَكُم مِّن ذُنُوبِكُمْ وَيُؤَخِّرَكُمْ إِلَىٰٓ أَجَلٍۢ مُّسَمًّۭى ۚ قَالُوٓا۟ إِنْ أَنتُمْ إِلَّا بَشَرٌۭ مِّثْلُنَا تُرِيدُونَ أَن تَصُدُّونَا عَمَّا كَانَ يَعْبُدُ ءَابَآؤُنَا فَأْتُونَا بِسُلْطَٰنٍۢ مُّبِينٍۢ</span>
<br><span class="ayah_translation">رسولان آنها گفتند: «آیا در خدا شکّ است؟! خدایی که آسمانها و زمین را آفریده؛ او شما را دعوت می‌کند تا گناهانتان را ببخشد، و تا موعد مقرّری شما را باقی گذارد!» آنها گفتند: «(ما اینها را نمی‌فهمیم! همین اندازه می‌دانیم که) شما انسانهایی همانند ما هستید، می‌خواهید ما را از آنچه پدرانمان می‌پرستیدند بازدارید؛ شما دلیل و معجزه روشنی برای ما بیاورید!»</span>

##### 11

<span class="ayah">قَالَتْ لَهُمْ رُسُلُهُمْ إِن نَّحْنُ إِلَّا بَشَرٌۭ مِّثْلُكُمْ وَلَٰكِنَّ ٱللَّهَ يَمُنُّ عَلَىٰ مَن يَشَآءُ مِنْ عِبَادِهِۦ ۖ وَمَا كَانَ لَنَآ أَن نَّأْتِيَكُم بِسُلْطَٰنٍ إِلَّا بِإِذْنِ ٱللَّهِ ۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ</span>
<br><span class="ayah_translation">پیامبرانشان به آنها گفتند: «درست است که ما بشری همانند شما هستیم، ولی خداوند بر هر کس از بندگانش بخواهد (و شایسته بداند)، نعمت می‌بخشد (و مقام رسالت عطا می‌کند)! و ما هرگز نمی‌توانیم معجزه‌ای جز بفرمان خدا بیاوریم! (و از تهدیدهای شما نمی‌هراسیم؛) افراد باایمان باید تنها بر خدا توکّل کنند!</span>

##### 12

<span class="ayah">وَمَا لَنَآ أَلَّا نَتَوَكَّلَ عَلَى ٱللَّهِ وَقَدْ هَدَىٰنَا سُبُلَنَا ۚ وَلَنَصْبِرَنَّ عَلَىٰ مَآ ءَاذَيْتُمُونَا ۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُتَوَكِّلُونَ</span>
<br><span class="ayah_translation">و چرا بر خدا توکّل نکنیم، با اینکه ما را به راه‌های (سعادت) رهبری کرده است؟! و ما بطور مسلّم در برابر آزارهای شما صبر خواهیم کرد (و دست از رسالت خویش بر نمی‌داریم)! و توکّل‌کنندگان، باید فقط بر خدا توکّل کنند!»</span>

##### 13

<span class="ayah">وَقَالَ ٱلَّذِينَ كَفَرُوا۟ لِرُسُلِهِمْ لَنُخْرِجَنَّكُم مِّنْ أَرْضِنَآ أَوْ لَتَعُودُنَّ فِى مِلَّتِنَا ۖ فَأَوْحَىٰٓ إِلَيْهِمْ رَبُّهُمْ لَنُهْلِكَنَّ ٱلظَّٰلِمِينَ</span>
<br><span class="ayah_translation">(ولی) کافران به پیامبران خود گفتند: «ما قطعاً شما را از سرزمین خود بیرون خواهیم کرد، مگر اینکه به آیین ما بازگردید!» در این حال، پروردگارشان به آنها وحی فرستاد که: «ما ظالمان را هلاک می‌کنیم!</span>

##### 14

<span class="ayah">وَلَنُسْكِنَنَّكُمُ ٱلْأَرْضَ مِنۢ بَعْدِهِمْ ۚ ذَٰلِكَ لِمَنْ خَافَ مَقَامِى وَخَافَ وَعِيدِ</span>
<br><span class="ayah_translation">و شما را بعد از آنان در زمین سکوت خواهیم داد، این (موفقیّت)، برای کسی است که از مقام (عدالت) من بترسد؛ و از عذاب (من) بیمناک باشد!»</span>

##### 15

<span class="ayah">وَٱسْتَفْتَحُوا۟ وَخَابَ كُلُّ جَبَّارٍ عَنِيدٍۢ</span>
<br><span class="ayah_translation">و آنها (از خدا) تقاضای فتح و پیروزی (بر کفار) کردند؛ و (سرانجام) هر گردنکش منحرفی نومید و نابود شد!</span>

##### 16

<span class="ayah">مِّن وَرَآئِهِۦ جَهَنَّمُ وَيُسْقَىٰ مِن مَّآءٍۢ صَدِيدٍۢ</span>
<br><span class="ayah_translation">به دنبال او جهنم خواهد بود؛ و از آب بد بوی متعفّنی نوشانده می‌شود!</span>

##### 17

<span class="ayah">يَتَجَرَّعُهُۥ وَلَا يَكَادُ يُسِيغُهُۥ وَيَأْتِيهِ ٱلْمَوْتُ مِن كُلِّ مَكَانٍۢ وَمَا هُوَ بِمَيِّتٍۢ ۖ وَمِن وَرَآئِهِۦ عَذَابٌ غَلِيظٌۭ</span>
<br><span class="ayah_translation">بزحمت جرعه جرعه آن را سرمی‌کشد؛ و هرگز حاضر نیست به میل خود آن را بیاشامد؛ و مرگ از هرجا به سراغ او می‌آید؛ ولی با این همه نمی‌میرد! و بدنبال آن، عذاب شدیدی است!</span>

##### 18

<span class="ayah">مَّثَلُ ٱلَّذِينَ كَفَرُوا۟ بِرَبِّهِمْ ۖ أَعْمَٰلُهُمْ كَرَمَادٍ ٱشْتَدَّتْ بِهِ ٱلرِّيحُ فِى يَوْمٍ عَاصِفٍۢ ۖ لَّا يَقْدِرُونَ مِمَّا كَسَبُوا۟ عَلَىٰ شَىْءٍۢ ۚ ذَٰلِكَ هُوَ ٱلضَّلَٰلُ ٱلْبَعِيدُ</span>
<br><span class="ayah_translation">اعمال کسانی که به پروردگارشان کافر شدند، همچون خاکستری است در برابر تندباد در یک روز طوفانی! آنها توانایی ندارند کمترین چیزی از آنچه را انجام داده‌اند، به دست آورند؛ و این همان گمراهی دور و دراز است!</span>

##### 19

<span class="ayah">أَلَمْ تَرَ أَنَّ ٱللَّهَ خَلَقَ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضَ بِٱلْحَقِّ ۚ إِن يَشَأْ يُذْهِبْكُمْ وَيَأْتِ بِخَلْقٍۢ جَدِيدٍۢ</span>
<br><span class="ayah_translation">آیا ندیدی خداوند، آسمانها و زمین را بحق آفریده است؟! اگر بخواهد، شما را می‌برد و خلق تازه‌ای می‌آورد!</span>

##### 20

<span class="ayah">وَمَا ذَٰلِكَ عَلَى ٱللَّهِ بِعَزِيزٍۢ</span>
<br><span class="ayah_translation">و این کار برای خدا مشکل نیست!</span>

##### 21

<span class="ayah">وَبَرَزُوا۟ لِلَّهِ جَمِيعًۭا فَقَالَ ٱلضُّعَفَٰٓؤُا۟ لِلَّذِينَ ٱسْتَكْبَرُوٓا۟ إِنَّا كُنَّا لَكُمْ تَبَعًۭا فَهَلْ أَنتُم مُّغْنُونَ عَنَّا مِنْ عَذَابِ ٱللَّهِ مِن شَىْءٍۢ ۚ قَالُوا۟ لَوْ هَدَىٰنَا ٱللَّهُ لَهَدَيْنَٰكُمْ ۖ سَوَآءٌ عَلَيْنَآ أَجَزِعْنَآ أَمْ صَبَرْنَا مَا لَنَا مِن مَّحِيصٍۢ</span>
<br><span class="ayah_translation">و (در قیامت)، همه آنها در برابر خدا ظاهر می‌شوند؛ در این هنگام، ضعفا [= دنباله‌روان نادان‌] به مستکبران (و رهبران گمراه) می‌گویند: «ما پیروان شما بودیم! آیا (اکنون که بخاطر پیروی از شما گرفتار مجازات الهی شده‌ایم،) شما حاضرید سهمی از عذاب الهی را بپذیرید و از ما بردارید؟» آنها می‌گویند: «اگر خدا ما را هدایت کرده بود، ما نیز شما را هدایت می‌کردیم! (ولی کار از اینها گذشته است،) چه بیتابی کنیم و چه شکیبایی، تفاوتی برای ما ندارد؛ راه گریزی برای ما نیست!»</span>

##### 22

<span class="ayah">وَقَالَ ٱلشَّيْطَٰنُ لَمَّا قُضِىَ ٱلْأَمْرُ إِنَّ ٱللَّهَ وَعَدَكُمْ وَعْدَ ٱلْحَقِّ وَوَعَدتُّكُمْ فَأَخْلَفْتُكُمْ ۖ وَمَا كَانَ لِىَ عَلَيْكُم مِّن سُلْطَٰنٍ إِلَّآ أَن دَعَوْتُكُمْ فَٱسْتَجَبْتُمْ لِى ۖ فَلَا تَلُومُونِى وَلُومُوٓا۟ أَنفُسَكُم ۖ مَّآ أَنَا۠ بِمُصْرِخِكُمْ وَمَآ أَنتُم بِمُصْرِخِىَّ ۖ إِنِّى كَفَرْتُ بِمَآ أَشْرَكْتُمُونِ مِن قَبْلُ ۗ إِنَّ ٱلظَّٰلِمِينَ لَهُمْ عَذَابٌ أَلِيمٌۭ</span>
<br><span class="ayah_translation">و شیطان، هنگامی که کار تمام می‌شود، می‌گوید: «خداوند به شما وعده حق داد؛ و من به شما وعده (باطل) دادم، و تخلّف کردم! من بر شما تسلّطی نداشتم، جز اینکه دعوتتان کردم و شما دعوت مرا پذیرفتید! بنابر این، مرا سرزنش نکنید؛ خود را سرزنش کنید! نه من فریادرس شما هستم، و نه شما فریادرس من! من نسبت به شرک شما درباره خود، که از قبل داشتید، (و اطاعت مرا همردیف اطاعت خدا قرار دادید) بیزار و کافرم!» مسلّماً ستمکاران عذاب دردناکی دارند!</span>

##### 23

<span class="ayah">وَأُدْخِلَ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ جَنَّٰتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَٰرُ خَٰلِدِينَ فِيهَا بِإِذْنِ رَبِّهِمْ ۖ تَحِيَّتُهُمْ فِيهَا سَلَٰمٌ</span>
<br><span class="ayah_translation">و کسانی را که ایمان آوردند و اعمال صالح انجام دادند، به باغهای بهشت وارد می‌کنند؛ باغهایی که نهرها از زیر درختانش جاری است؛ به اذن پروردگارشان، جاودانه در آن می‌مانند؛ و تحیّت آنها در آن، «سلام» است.</span>

##### 24

<span class="ayah">أَلَمْ تَرَ كَيْفَ ضَرَبَ ٱللَّهُ مَثَلًۭا كَلِمَةًۭ طَيِّبَةًۭ كَشَجَرَةٍۢ طَيِّبَةٍ أَصْلُهَا ثَابِتٌۭ وَفَرْعُهَا فِى ٱلسَّمَآءِ</span>
<br><span class="ayah_translation">آیا ندیدی چگونه خداوند «کلمه طیبه» (و گفتار پاکیزه) را به درخت پاکیزه‌ای تشبیه کرده که ریشه آن (در زمین) ثابت، و شاخه آن در آسمان است؟!</span>

##### 25

<span class="ayah">تُؤْتِىٓ أُكُلَهَا كُلَّ حِينٍۭ بِإِذْنِ رَبِّهَا ۗ وَيَضْرِبُ ٱللَّهُ ٱلْأَمْثَالَ لِلنَّاسِ لَعَلَّهُمْ يَتَذَكَّرُونَ</span>
<br><span class="ayah_translation">هر زمان میوه خود را به اذن پروردگارش می‌دهد. و خداوند برای مردم مثلها می‌زند، شاید متذکّر شوند (و پند گیرند)!</span>

##### 26

<span class="ayah">وَمَثَلُ كَلِمَةٍ خَبِيثَةٍۢ كَشَجَرَةٍ خَبِيثَةٍ ٱجْتُثَّتْ مِن فَوْقِ ٱلْأَرْضِ مَا لَهَا مِن قَرَارٍۢ</span>
<br><span class="ayah_translation">(همچنین) «کلمه خبیثه» (و سخن آلوده) را به درخت ناپاکی تشبیه کرده که از روی زمین برکنده شده، و قرار و ثباتی ندارد.</span>

##### 27

<span class="ayah">يُثَبِّتُ ٱللَّهُ ٱلَّذِينَ ءَامَنُوا۟ بِٱلْقَوْلِ ٱلثَّابِتِ فِى ٱلْحَيَوٰةِ ٱلدُّنْيَا وَفِى ٱلْءَاخِرَةِ ۖ وَيُضِلُّ ٱللَّهُ ٱلظَّٰلِمِينَ ۚ وَيَفْعَلُ ٱللَّهُ مَا يَشَآءُ</span>
<br><span class="ayah_translation">خداوند کسانی را که ایمان آوردند، به خاطر گفتار و اعتقاد ثابتشان، استوار می‌دارد؛ هم در این جهان، و هم در سرای دیگر! و ستمگران را گمراه می‌سازد، (و لطف خود را از آنها برمی‌گیرد)؛ خداوند هر کار را بخواهد (و مصلحت بداند) انجام می‌دهد!</span>

##### 28

<span class="ayah">۞ أَلَمْ تَرَ إِلَى ٱلَّذِينَ بَدَّلُوا۟ نِعْمَتَ ٱللَّهِ كُفْرًۭا وَأَحَلُّوا۟ قَوْمَهُمْ دَارَ ٱلْبَوَارِ</span>
<br><span class="ayah_translation">آیا ندیدی کسانی را که نعمت خدا را به کفران تبدیل کردند، و قوم خود را به سرای نیستی و نابودی کشاندند؟!</span>

##### 29

<span class="ayah">جَهَنَّمَ يَصْلَوْنَهَا ۖ وَبِئْسَ ٱلْقَرَارُ</span>
<br><span class="ayah_translation">(سرای نیستی و نابودی، همان) جهنم است که آنها در آتش آن وارد می‌شوند؛ و بد قرارگاهی است!</span>

##### 30

<span class="ayah">وَجَعَلُوا۟ لِلَّهِ أَندَادًۭا لِّيُضِلُّوا۟ عَن سَبِيلِهِۦ ۗ قُلْ تَمَتَّعُوا۟ فَإِنَّ مَصِيرَكُمْ إِلَى ٱلنَّارِ</span>
<br><span class="ayah_translation">آنها برای خدا همتایانی قرار داده‌اند، تا (مردم را) از راه او (منحرف و) گمراه سازند؛ بگو: «(چند روزی از زندگی دنیا و لذّات آن) بهره گیرید؛ امّا عاقبت کار شما به سوی آتش (دوزخ) است!»</span>

##### 31

<span class="ayah">قُل لِّعِبَادِىَ ٱلَّذِينَ ءَامَنُوا۟ يُقِيمُوا۟ ٱلصَّلَوٰةَ وَيُنفِقُوا۟ مِمَّا رَزَقْنَٰهُمْ سِرًّۭا وَعَلَانِيَةًۭ مِّن قَبْلِ أَن يَأْتِىَ يَوْمٌۭ لَّا بَيْعٌۭ فِيهِ وَلَا خِلَٰلٌ</span>
<br><span class="ayah_translation">به بندگان من که ایمان آورده‌اند بگو نماز را برپا دارند؛ و از آنچه به آنها روزی داده‌ایم، پنهان و آشکار، انفاق کنند؛ پیش از آنکه روزی فرا رسد که نه در آن خرید و فروش است، و نه دوستی! (نه با مال می‌توانند از کیفر خدا رهایی یابند، و نه با پیوندهای مادی!)</span>

##### 32

<span class="ayah">ٱللَّهُ ٱلَّذِى خَلَقَ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضَ وَأَنزَلَ مِنَ ٱلسَّمَآءِ مَآءًۭ فَأَخْرَجَ بِهِۦ مِنَ ٱلثَّمَرَٰتِ رِزْقًۭا لَّكُمْ ۖ وَسَخَّرَ لَكُمُ ٱلْفُلْكَ لِتَجْرِىَ فِى ٱلْبَحْرِ بِأَمْرِهِۦ ۖ وَسَخَّرَ لَكُمُ ٱلْأَنْهَٰرَ</span>
<br><span class="ayah_translation">خداوند همان کسی است که آسمانها و زمین را آفرید؛ و از آسمان، آبی نازل کرد؛ و با آن، میوه‌ها (ی مختلف) را برای روزی شما (از زمین) بیرون آورد؛ و کشتی‌ها را مسخّر شما گردانید، تا بر صفحه دریا به فرمان او حرکت کنند؛ و نهرها را (نیز) مسخّر شما نمود؛</span>

##### 33

<span class="ayah">وَسَخَّرَ لَكُمُ ٱلشَّمْسَ وَٱلْقَمَرَ دَآئِبَيْنِ ۖ وَسَخَّرَ لَكُمُ ٱلَّيْلَ وَٱلنَّهَارَ</span>
<br><span class="ayah_translation">و خورشید و ماه را -که با برنامه منظّمی درکارند- به تسخیر شما درآورد؛ و شب و روز را (نیز) مسخّر شما ساخت؛</span>

##### 34

<span class="ayah">وَءَاتَىٰكُم مِّن كُلِّ مَا سَأَلْتُمُوهُ ۚ وَإِن تَعُدُّوا۟ نِعْمَتَ ٱللَّهِ لَا تُحْصُوهَآ ۗ إِنَّ ٱلْإِنسَٰنَ لَظَلُومٌۭ كَفَّارٌۭ</span>
<br><span class="ayah_translation">و از هر چیزی که از او خواستید، به شما داد؛ و اگر نعمتهای خدا را بشمارید، هرگز آنها را شماره نتوانید کرد! انسان، ستمگر و ناسپاس است!</span>

##### 35

<span class="ayah">وَإِذْ قَالَ إِبْرَٰهِيمُ رَبِّ ٱجْعَلْ هَٰذَا ٱلْبَلَدَ ءَامِنًۭا وَٱجْنُبْنِى وَبَنِىَّ أَن نَّعْبُدَ ٱلْأَصْنَامَ</span>
<br><span class="ayah_translation">(به یاد آورید) زمانی را که ابراهیم گفت: «پروردگارا! این شهر [= مکّه‌] را شهر امنی قرار ده! و من و فرزندانم را از پرستش بتها دور نگاه دار!</span>

##### 36

<span class="ayah">رَبِّ إِنَّهُنَّ أَضْلَلْنَ كَثِيرًۭا مِّنَ ٱلنَّاسِ ۖ فَمَن تَبِعَنِى فَإِنَّهُۥ مِنِّى ۖ وَمَنْ عَصَانِى فَإِنَّكَ غَفُورٌۭ رَّحِيمٌۭ</span>
<br><span class="ayah_translation">پروردگارا! آنها [= بتها] بسیاری از مردم را گمراه ساختند! هر کس از من پیروی کند از من است؛ و هر کس نافرمانی من کند، تو بخشنده و مهربانی!</span>

##### 37

<span class="ayah">رَّبَّنَآ إِنِّىٓ أَسْكَنتُ مِن ذُرِّيَّتِى بِوَادٍ غَيْرِ ذِى زَرْعٍ عِندَ بَيْتِكَ ٱلْمُحَرَّمِ رَبَّنَا لِيُقِيمُوا۟ ٱلصَّلَوٰةَ فَٱجْعَلْ أَفْـِٔدَةًۭ مِّنَ ٱلنَّاسِ تَهْوِىٓ إِلَيْهِمْ وَٱرْزُقْهُم مِّنَ ٱلثَّمَرَٰتِ لَعَلَّهُمْ يَشْكُرُونَ</span>
<br><span class="ayah_translation">پروردگارا! من بعضی از فرزندانم را در سرزمین بی‌آب و علفی، در کنار خانه‌ای که حرم توست، ساکن ساختم تا نماز را برپا دارند؛ تو دلهای گروهی از مردم را متوجّه آنها ساز؛ و از ثمرات به آنها روزی ده؛ شاید آنان شکر تو را بجای آورند!</span>

##### 38

<span class="ayah">رَبَّنَآ إِنَّكَ تَعْلَمُ مَا نُخْفِى وَمَا نُعْلِنُ ۗ وَمَا يَخْفَىٰ عَلَى ٱللَّهِ مِن شَىْءٍۢ فِى ٱلْأَرْضِ وَلَا فِى ٱلسَّمَآءِ</span>
<br><span class="ayah_translation">پروردگارا! تو می‌دانی آنچه را ما پنهان و یا آشکار می‌کنیم؛ و چیزی در زمین و آسمان بر خدا پنهان نیست!</span>

##### 39

<span class="ayah">ٱلْحَمْدُ لِلَّهِ ٱلَّذِى وَهَبَ لِى عَلَى ٱلْكِبَرِ إِسْمَٰعِيلَ وَإِسْحَٰقَ ۚ إِنَّ رَبِّى لَسَمِيعُ ٱلدُّعَآءِ</span>
<br><span class="ayah_translation">حمد خدای را که در پیری، اسماعیل و اسحاق را به من بخشید؛ مسلّماً پروردگار من، شنونده (و اجابت کننده) دعاست.</span>

##### 40

<span class="ayah">رَبِّ ٱجْعَلْنِى مُقِيمَ ٱلصَّلَوٰةِ وَمِن ذُرِّيَّتِى ۚ رَبَّنَا وَتَقَبَّلْ دُعَآءِ</span>
<br><span class="ayah_translation">پروردگارا: مرا برپا کننده نماز قرار ده، و از فرزندانم (نیز چنین فرما)، پروردگارا: دعای مرا بپذیر!</span>

##### 41

<span class="ayah">رَبَّنَا ٱغْفِرْ لِى وَلِوَٰلِدَىَّ وَلِلْمُؤْمِنِينَ يَوْمَ يَقُومُ ٱلْحِسَابُ</span>
<br><span class="ayah_translation">پروردگارا! من و پدر و مادرم و همه مؤمنان را، در آن روز که حساب برپا می‌شود، بیامرز!</span>

##### 42

<span class="ayah">وَلَا تَحْسَبَنَّ ٱللَّهَ غَٰفِلًا عَمَّا يَعْمَلُ ٱلظَّٰلِمُونَ ۚ إِنَّمَا يُؤَخِّرُهُمْ لِيَوْمٍۢ تَشْخَصُ فِيهِ ٱلْأَبْصَٰرُ</span>
<br><span class="ayah_translation">گمان مبر که خدا، از آنچه ظالمان انجام می‌دهند، غافل است! (نه، بلکه کیفر) آنها را برای روزی تأخیر انداخته است که چشمها در آن (به خاطر ترس و وحشت) از حرکت بازمی‌ایستد...</span>

##### 43

<span class="ayah">مُهْطِعِينَ مُقْنِعِى رُءُوسِهِمْ لَا يَرْتَدُّ إِلَيْهِمْ طَرْفُهُمْ ۖ وَأَفْـِٔدَتُهُمْ هَوَآءٌۭ</span>
<br><span class="ayah_translation">گردنها را کشیده، سرها را به آسمان بلندکرده، حتّی پلک چشمهایشان از حرکت بازمی‌ماند؛ (زیرا به هر طرف نگاه کنند، آثار عذاب آشکار است!) و (در این حال) دلهایشان (فرومی‌ریزد؛ و از اندیشه و امید،) خالی می‌گردد!</span>

##### 44

<span class="ayah">وَأَنذِرِ ٱلنَّاسَ يَوْمَ يَأْتِيهِمُ ٱلْعَذَابُ فَيَقُولُ ٱلَّذِينَ ظَلَمُوا۟ رَبَّنَآ أَخِّرْنَآ إِلَىٰٓ أَجَلٍۢ قَرِيبٍۢ نُّجِبْ دَعْوَتَكَ وَنَتَّبِعِ ٱلرُّسُلَ ۗ أَوَلَمْ تَكُونُوٓا۟ أَقْسَمْتُم مِّن قَبْلُ مَا لَكُم مِّن زَوَالٍۢ</span>
<br><span class="ayah_translation">و مردم را از روزی که عذاب الهی به سراغشان می‌آید، بترسان! آن روز که ظالمان می‌گویند: «پروردگارا! مدّت کوتاهی ما را مهلت ده، تا دعوت تو را بپذیریم و از پیامبران پیروی کنیم!» (امّا پاسخ می‌شنوند که:) مگر قبلاً سوگند یاد نکرده بودید که زوال و فنایی برای شما نیست؟!</span>

##### 45

<span class="ayah">وَسَكَنتُمْ فِى مَسَٰكِنِ ٱلَّذِينَ ظَلَمُوٓا۟ أَنفُسَهُمْ وَتَبَيَّنَ لَكُمْ كَيْفَ فَعَلْنَا بِهِمْ وَضَرَبْنَا لَكُمُ ٱلْأَمْثَالَ</span>
<br><span class="ayah_translation">(آری شما بودید که) در منازل (و کاخهای) کسانی که به خویشتن ستم کردند، ساکن شدید؛ و برای شما آشکار شد چگونه با آنان رفتار کردیم؛ و برای شما، مثلها (از سرگذشت پیشینیان) زدیم (باز هم بیدار نشدید)!</span>

##### 46

<span class="ayah">وَقَدْ مَكَرُوا۟ مَكْرَهُمْ وَعِندَ ٱللَّهِ مَكْرُهُمْ وَإِن كَانَ مَكْرُهُمْ لِتَزُولَ مِنْهُ ٱلْجِبَالُ</span>
<br><span class="ayah_translation">آنها نهایت مکر (و نیرنگ) خود را به کار زدند؛ و همه مکرها (و توطئه‌هایشان) نزد خدا آشکار است، هر چند مکرشان چنان باشد که کوه‌ها را از جا برکند!</span>

##### 47

<span class="ayah">فَلَا تَحْسَبَنَّ ٱللَّهَ مُخْلِفَ وَعْدِهِۦ رُسُلَهُۥٓ ۗ إِنَّ ٱللَّهَ عَزِيزٌۭ ذُو ٱنتِقَامٍۢ</span>
<br><span class="ayah_translation">پس گمان مبر که خدا وعده‌ای را که به پیامبرانش داده، تخلّف کند! چرا که خداوند قادر و انتقام گیرنده است.</span>

##### 48

<span class="ayah">يَوْمَ تُبَدَّلُ ٱلْأَرْضُ غَيْرَ ٱلْأَرْضِ وَٱلسَّمَٰوَٰتُ ۖ وَبَرَزُوا۟ لِلَّهِ ٱلْوَٰحِدِ ٱلْقَهَّارِ</span>
<br><span class="ayah_translation">در آن روز که این زمین به زمین دیگر، و آسمانها (به آسمانهای دیگری) مبدل می‌شود، و آنان در پیشگاه خداوند واحد قهار ظاهر می‌گردند!</span>

##### 49

<span class="ayah">وَتَرَى ٱلْمُجْرِمِينَ يَوْمَئِذٍۢ مُّقَرَّنِينَ فِى ٱلْأَصْفَادِ</span>
<br><span class="ayah_translation">و در آن روز، مجرمان را با هم در غل و زنجیر می‌بینی! (که دستها و گردنهایشان را به هم بسته است!)</span>

##### 50

<span class="ayah">سَرَابِيلُهُم مِّن قَطِرَانٍۢ وَتَغْشَىٰ وُجُوهَهُمُ ٱلنَّارُ</span>
<br><span class="ayah_translation">لباسهایشان از قطران [= ماده چسبنده بد بوی قابل اشتعال‌] است؛ و صورتهایشان را آتش می‌پوشاند...</span>

##### 51

<span class="ayah">لِيَجْزِىَ ٱللَّهُ كُلَّ نَفْسٍۢ مَّا كَسَبَتْ ۚ إِنَّ ٱللَّهَ سَرِيعُ ٱلْحِسَابِ</span>
<br><span class="ayah_translation">تا خداوند هر کس را، هر آنچه انجام داده، جزا دهد! به یقین، خداوند سریع الحساب است!</span>

##### 52

<span class="ayah">هَٰذَا بَلَٰغٌۭ لِّلنَّاسِ وَلِيُنذَرُوا۟ بِهِۦ وَلِيَعْلَمُوٓا۟ أَنَّمَا هُوَ إِلَٰهٌۭ وَٰحِدٌۭ وَلِيَذَّكَّرَ أُو۟لُوا۟ ٱلْأَلْبَٰبِ</span>
<br><span class="ayah_translation">این (قرآن،) پیام (و ابلاغی) برای (عموم) مردم است؛ تا همه به وسیله آن انذار شوند، و بدانند او خدا یکتاست؛ و تا صاحبان مغز (و اندیشه) پند گیرند!</span>
