##### 2
<span class="ayah hovertext" data-hover="تو -به لطف پروردگارتديوانه نيستى‌">مَآ أَنتَ بِنِعْمَةِ رَبِّكَ بِمَجْنُونٍۢ</span>
##### 3
<span class="ayah hovertext" data-hover="و تو را پاداشى ناكاسته [/بى‌منت‌] است‌">وَإِنَّ لَكَ لَأَجْرًا غَيْرَ مَمْنُونٍۢ</span>
##### 4
<span class="ayah hovertext" data-hover="و تو داراى اخلاق بسيار بزرگوارانه‌اى هستى‌">وَإِنَّكَ لَعَلَىٰ خُلُقٍ عَظِيمٍۢ</span>
##### 5
<span class="ayah hovertext" data-hover="زودا كه بنگرى و بنگرند">فَسَتُبْصِرُ وَيُبْصِرُونَ</span>
##### 6
<span class="ayah hovertext" data-hover="كه كداميك از شما ديوانه‌ايد">بِأَييِّكُمُ ٱلْمَفْتُونُ</span>
##### 7
<span class="ayah hovertext" data-hover="بى‌گمان پروردگارت به كسى كه از راه او گم گشته و به رهيافتگان آگاه‌تر است‌">إِنَّ رَبَّكَ هُوَ أَعْلَمُ بِمَن ضَلَّ عَن سَبِيلِهِۦ وَهُوَ أَعْلَمُ بِٱلْمُهْتَدِينَ</span>
##### 8
<span class="ayah hovertext" data-hover="پس از منكران اطاعت مكن‌">فَلَا تُطِعِ ٱلْمُكَذِّبِينَ</span>
##### 9
<span class="ayah hovertext" data-hover="خوش دارند كه با آنان سازگارى كنى تا با تو سازگارى كنند">وَدُّوا۟ لَوْ تُدْهِنُ فَيُدْهِنُونَ</span>
##### 10
<span class="ayah hovertext" data-hover="و از هيچ بيهوده سوگند خوار پستى اطاعت مكن‌">وَلَا تُطِعْ كُلَّ حَلَّافٍۢ مَّهِينٍ</span>
##### 11
<span class="ayah hovertext" data-hover="عيبجوى روانه در پى سخن‌چينى‌">هَمَّازٍۢ مَّشَّآءٍۭ بِنَمِيمٍۢ</span>
##### 12
<span class="ayah hovertext" data-hover="باز دارنده، از نيكى، تجاوزگر گناهكار">مَّنَّاعٍۢ لِّلْخَيْرِ مُعْتَدٍ أَثِيمٍ</span>
##### 13
<span class="ayah hovertext" data-hover="درشت‌خوى، و بعد از اين همه، ناپاكزاد">عُتُلٍّۭ بَعْدَ ذَٰلِكَ زَنِيمٍ</span>
##### 14
<span class="ayah hovertext" data-hover="به خاطر آنكه مالدار است و صاحب پسران‌">أَن كَانَ ذَا مَالٍۢ وَبَنِينَ</span>
##### 15
<span class="ayah hovertext" data-hover="چون آيات ما بر او خوانده شود، گويد افسانه‌هاى پيشينيان است‌">إِذَا تُتْلَىٰ عَلَيْهِ ءَايَٰتُنَا قَالَ أَسَٰطِيرُ ٱلْأَوَّلِينَ</span>
##### 16
<span class="ayah hovertext" data-hover="زودا كه بر بينى او نشانه‌اى بگذاريم‌">سَنَسِمُهُۥ عَلَى ٱلْخُرْطُومِ</span>
##### 17
<span class="ayah hovertext" data-hover="ما آنان را آزموده‌ايم، همچنانكه باغداران را هم آزموده‌ايم، آنگاه كه سوگند خوردند كه صبحگاهان ميوه‌هاى آن را بازچينند">إِنَّا بَلَوْنَٰهُمْ كَمَا بَلَوْنَآ أَصْحَٰبَ ٱلْجَنَّةِ إِذْ أَقْسَمُوا۟ لَيَصْرِمُنَّهَا مُصْبِحِينَ</span>
##### 18
<span class="ayah hovertext" data-hover="و ان شاء الله نگفتند">وَلَا يَسْتَثْنُونَ</span>
##### 19
<span class="ayah hovertext" data-hover="سپس بلايى شبانه از جانب پروردگارت، در حالى كه ايشان خفته بودند، بر آن زد">فَطَافَ عَلَيْهَا طَآئِفٌۭ مِّن رَّبِّكَ وَهُمْ نَآئِمُونَ</span>
##### 20
<span class="ayah hovertext" data-hover="و مانند خاكستر سياه شد">فَأَصْبَحَتْ كَٱلصَّرِيمِ</span>
##### 21
<span class="ayah hovertext" data-hover="[آنان بى‌خبر] صبحگاهان همديگر را فراخواندند">فَتَنَادَوْا۟ مُصْبِحِينَ</span>
##### 22
<span class="ayah hovertext" data-hover="كه اگر ميوه‌چين هستيد، پگاه به سراغ كشتزارتان برويد">أَنِ ٱغْدُوا۟ عَلَىٰ حَرْثِكُمْ إِن كُنتُمْ صَٰرِمِينَ</span>
##### 23
<span class="ayah hovertext" data-hover="سپس رهسپار شدند، در حالى كه آهسته با همديگر سخن مى‌گفتند">فَٱنطَلَقُوا۟ وَهُمْ يَتَخَٰفَتُونَ</span>
##### 24
<span class="ayah hovertext" data-hover="كه مبادا بى‌نوايى به آنجا بر شما وارد شود">أَن لَّا يَدْخُلَنَّهَا ٱلْيَوْمَ عَلَيْكُم مِّسْكِينٌۭ</span>
##### 25
<span class="ayah hovertext" data-hover="و بر منع و مضايقه توانا شدند">وَغَدَوْا۟ عَلَىٰ حَرْدٍۢ قَٰدِرِينَ</span>
##### 26
<span class="ayah hovertext" data-hover="و چون آن [باغ سياه‌سوخته‌] را ديدند [جا خوردند و] گفتند ما گم گشته‌ايم‌">فَلَمَّا رَأَوْهَا قَالُوٓا۟ إِنَّا لَضَآلُّونَ</span>
##### 27
<span class="ayah hovertext" data-hover="نه، مثل اينكه، بى‌بهره‌ايم‌">بَلْ نَحْنُ مَحْرُومُونَ</span>
##### 28
<span class="ayah hovertext" data-hover="بهترينشان گفت مگر به شما نگفته بودم كه چرا تسبيح [و ان شاء الله‌] نمى‌گوييد؟">قَالَ أَوْسَطُهُمْ أَلَمْ أَقُل لَّكُمْ لَوْلَا تُسَبِّحُونَ</span>
##### 29
<span class="ayah hovertext" data-hover="گفتند پاك است پروردگارمان، ما خود ستمكار بوديم‌">قَالُوا۟ سُبْحَٰنَ رَبِّنَآ إِنَّا كُنَّا ظَٰلِمِينَ</span>
##### 30
<span class="ayah hovertext" data-hover="سپس به سرزنش به همديگر روى آوردند">فَأَقْبَلَ بَعْضُهُمْ عَلَىٰ بَعْضٍۢ يَتَلَٰوَمُونَ</span>
##### 31
<span class="ayah hovertext" data-hover="گفتند واى بر ما، ما سركش بوده‌ايم‌">قَالُوا۟ يَٰوَيْلَنَآ إِنَّا كُنَّا طَٰغِينَ</span>
##### 32
<span class="ayah hovertext" data-hover="چه بسا پروردگارمان بهتر از آن را برايمان جانشين گرداند، ما به سوى پروردگارمان مى‌گراييم‌">عَسَىٰ رَبُّنَآ أَن يُبْدِلَنَا خَيْرًۭا مِّنْهَآ إِنَّآ إِلَىٰ رَبِّنَا رَٰغِبُونَ</span>
##### 33
<span class="ayah hovertext" data-hover="اين چنين است عذاب، و اگر مى‌دانستند عذاب اخروى سهمگين‌تر است‌">كَذَٰلِكَ ٱلْعَذَابُ ۖ وَلَعَذَابُ ٱلْءَاخِرَةِ أَكْبَرُ ۚ لَوْ كَانُوا۟ يَعْلَمُونَ</span>
##### 34
<span class="ayah hovertext" data-hover="براى پرهيزگاران در نزد پروردگارشان بهشتهاى پرناز و نعمت است‌">إِنَّ لِلْمُتَّقِينَ عِندَ رَبِّهِمْ جَنَّٰتِ ٱلنَّعِيمِ</span>
##### 35
<span class="ayah hovertext" data-hover="آيا مسلمانان را همانند گناهكاران مى‌شماريم؟">أَفَنَجْعَلُ ٱلْمُسْلِمِينَ كَٱلْمُجْرِمِينَ</span>
##### 36
<span class="ayah hovertext" data-hover="شما را چه مى‌شود چگونه داورى مى‌كنيد؟">مَا لَكُمْ كَيْفَ تَحْكُمُونَ</span>
##### 37
<span class="ayah hovertext" data-hover="يا مگر كتابى داريد كه در آنها مطالعه مى‌كنيد">أَمْ لَكُمْ كِتَٰبٌۭ فِيهِ تَدْرُسُونَ</span>
##### 38
<span class="ayah hovertext" data-hover="كه در آن براى شما هرچه آرزو كنيد هست؟">إِنَّ لَكُمْ فِيهِ لَمَا تَخَيَّرُونَ</span>
##### 39
<span class="ayah hovertext" data-hover="يا شما را پيمانهايى است با ما كه حكم آن تا قيامت باقى باشد كه هر چه حكم كنيد براى شما باشد؟">أَمْ لَكُمْ أَيْمَٰنٌ عَلَيْنَا بَٰلِغَةٌ إِلَىٰ يَوْمِ ٱلْقِيَٰمَةِ ۙ إِنَّ لَكُمْ لَمَا تَحْكُمُونَ</span>
##### 40
<span class="ayah hovertext" data-hover="از ايشان بپرس كه كداميك از آنان ضامن اين امر است؟">سَلْهُمْ أَيُّهُم بِذَٰلِكَ زَعِيمٌ</span>
##### 41
<span class="ayah hovertext" data-hover="يا شركايى دارند، پس اگر راست مى‌گويند شريكانشان را به ميان آورند">أَمْ لَهُمْ شُرَكَآءُ فَلْيَأْتُوا۟ بِشُرَكَآئِهِمْ إِن كَانُوا۟ صَٰدِقِينَ</span>
##### 42
<span class="ayah hovertext" data-hover="روزى كه هنگامه بالا گيرد، و آنان را به سجده دعوت كنند ولى نتوانند">يَوْمَ يُكْشَفُ عَن سَاقٍۢ وَيُدْعَوْنَ إِلَى ٱلسُّجُودِ فَلَا يَسْتَطِيعُونَ</span>
##### 43
<span class="ayah hovertext" data-hover="ديدگانشان [از ترس‌] فرو افتاده باشد، غبار خفت و خوارى بر رخسارشان نشسته، و چنين بود كه به سجده دعوت مى‌شدند و تندرست بودند">خَٰشِعَةً أَبْصَٰرُهُمْ تَرْهَقُهُمْ ذِلَّةٌۭ ۖ وَقَدْ كَانُوا۟ يُدْعَوْنَ إِلَى ٱلسُّجُودِ وَهُمْ سَٰلِمُونَ</span>
##### 44
<span class="ayah hovertext" data-hover="پس مرا با كسى كه اين سخن را تكذيب مى‌كند، وابگذار، زودا كه آهسته آهسته به نحوى كه در نيابند، فرو خواهيمشان گرفت‌">فَذَرْنِى وَمَن يُكَذِّبُ بِهَٰذَا ٱلْحَدِيثِ ۖ سَنَسْتَدْرِجُهُم مِّنْ حَيْثُ لَا يَعْلَمُونَ</span>
##### 45
<span class="ayah hovertext" data-hover="و مهلتشان مى‌دهم، زيرا تدبير من سخت استوار است‌">وَأُمْلِى لَهُمْ ۚ إِنَّ كَيْدِى مَتِينٌ</span>
##### 46
<span class="ayah hovertext" data-hover="يا مگر از آنان مزدى درخواست مى‌كنى، و آنان خود را زير بار تاوان، گرانبار مى‌يابند؟">أَمْ تَسْـَٔلُهُمْ أَجْرًۭا فَهُم مِّن مَّغْرَمٍۢ مُّثْقَلُونَ</span>
##### 47
<span class="ayah hovertext" data-hover="يا مگر علم و اسرار غيب با ايشان است و از روى آن مى‌نويسند؟">أَمْ عِندَهُمُ ٱلْغَيْبُ فَهُمْ يَكْتُبُونَ</span>
##### 48
<span class="ayah hovertext" data-hover="پس در انتظار حكم پروردگارت شكيبايى كن و همانند صاحب ماهى [ يونس‌] مباش كه [در تاريكى‌] ندا در داد و خشم فرو خورده بود">فَٱصْبِرْ لِحُكْمِ رَبِّكَ وَلَا تَكُن كَصَاحِبِ ٱلْحُوتِ إِذْ نَادَىٰ وَهُوَ مَكْظُومٌۭ</span>
##### 49
<span class="ayah hovertext" data-hover="اگر نعمتى از جانب پروردگارش او را دستگيرى نمى‌كرد، به كرانه‌[ى بى‌آب و علف‌] افكنده شده بود و قابل ملامت بود">لَّوْلَآ أَن تَدَٰرَكَهُۥ نِعْمَةٌۭ مِّن رَّبِّهِۦ لَنُبِذَ بِٱلْعَرَآءِ وَهُوَ مَذْمُومٌۭ</span>
##### 50
<span class="ayah hovertext" data-hover="سپس پروردگارش او را برگزيد و از شايستگانش قرار داد">فَٱجْتَبَٰهُ رَبُّهُۥ فَجَعَلَهُۥ مِنَ ٱلصَّٰلِحِينَ</span>
##### 51
<span class="ayah hovertext" data-hover="و بسيار نزديك بود كه كافران چون قرآن را شنيدند، تو را با ديدگانشان آسيب برسانند و گفتند او ديوانه است‌">وَإِن يَكَادُ ٱلَّذِينَ كَفَرُوا۟ لَيُزْلِقُونَكَ بِأَبْصَٰرِهِمْ لَمَّا سَمِعُوا۟ ٱلذِّكْرَ وَيَقُولُونَ إِنَّهُۥ لَمَجْنُونٌۭ</span>
