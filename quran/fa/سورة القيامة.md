▶ [[سورة المدثر]] | [[سورة الإنسان]] ◀

<br>

##### 1

<span class="ayah">بِسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ لَآ أُقْسِمُ بِيَوْمِ ٱلْقِيَٰمَةِ</span>
<br><span class="ayah_translation">سوگند به روز قیامت،</span>

##### 2

<span class="ayah">وَلَآ أُقْسِمُ بِٱلنَّفْسِ ٱللَّوَّامَةِ</span>
<br><span class="ayah_translation">و سوگند به (نفس لوّامه و) وجدان بیدار و ملامتگر (که رستاخیز حقّ است)!</span>

##### 3

<span class="ayah">أَيَحْسَبُ ٱلْإِنسَٰنُ أَلَّن نَّجْمَعَ عِظَامَهُۥ</span>
<br><span class="ayah_translation">آیا انسان می‌پندارد که هرگز استخوانهای او را جمع نخواهیم کرد؟!</span>

##### 4

<span class="ayah">بَلَىٰ قَٰدِرِينَ عَلَىٰٓ أَن نُّسَوِّىَ بَنَانَهُۥ</span>
<br><span class="ayah_translation">آری قادریم که (حتّی خطوط سر) انگشتان او را موزون و مرتّب کنیم!</span>

##### 5

<span class="ayah">بَلْ يُرِيدُ ٱلْإِنسَٰنُ لِيَفْجُرَ أَمَامَهُۥ</span>
<br><span class="ayah_translation">(انسان شک در معاد ندارد) بلکه او می‌خواهد (آزاد باشد و بدون ترس از دادگاه قیامت) در تمام عمر گناه کند!</span>

##### 6

<span class="ayah">يَسْـَٔلُ أَيَّانَ يَوْمُ ٱلْقِيَٰمَةِ</span>
<br><span class="ayah_translation">(از این رو) می‌پرسد: «قیامت کی خواهد بود»!</span>

##### 7

<span class="ayah">فَإِذَا بَرِقَ ٱلْبَصَرُ</span>
<br><span class="ayah_translation">(بگو:) در آن هنگام که چشمها از شدّت وحشت به گردش درآید،</span>

##### 8

<span class="ayah">وَخَسَفَ ٱلْقَمَرُ</span>
<br><span class="ayah_translation">و ماه بی‌نور گردد،</span>

##### 9

<span class="ayah">وَجُمِعَ ٱلشَّمْسُ وَٱلْقَمَرُ</span>
<br><span class="ayah_translation">و خورشید و ماه یک جا جمع شوند،</span>

##### 10

<span class="ayah">يَقُولُ ٱلْإِنسَٰنُ يَوْمَئِذٍ أَيْنَ ٱلْمَفَرُّ</span>
<br><span class="ayah_translation">آن روز انسان می‌گوید: «راه فرار کجاست؟!»</span>

##### 11

<span class="ayah">كَلَّا لَا وَزَرَ</span>
<br><span class="ayah_translation">هرگز چنین نیست، راه فرار و پناهگاهی وجود ندارد!</span>

##### 12

<span class="ayah">إِلَىٰ رَبِّكَ يَوْمَئِذٍ ٱلْمُسْتَقَرُّ</span>
<br><span class="ayah_translation">آن روز قرارگاه نهایی تنها بسوی پروردگار تو است؛</span>

##### 13

<span class="ayah">يُنَبَّؤُا۟ ٱلْإِنسَٰنُ يَوْمَئِذٍۭ بِمَا قَدَّمَ وَأَخَّرَ</span>
<br><span class="ayah_translation">و در آن روز انسان را از تمام کارهایی که از پیش یا پس فرستاده آگاه می‌کنند!</span>

##### 14

<span class="ayah">بَلِ ٱلْإِنسَٰنُ عَلَىٰ نَفْسِهِۦ بَصِيرَةٌۭ</span>
<br><span class="ayah_translation">بلکه انسان خودش از وضع خود آگاه است،</span>

##### 15

<span class="ayah">وَلَوْ أَلْقَىٰ مَعَاذِيرَهُۥ</span>
<br><span class="ayah_translation">هر چند (در ظاهر) برای خود عذرهایی بتراشد!</span>

##### 16

<span class="ayah">لَا تُحَرِّكْ بِهِۦ لِسَانَكَ لِتَعْجَلَ بِهِۦٓ</span>
<br><span class="ayah_translation">زبانت را بخاطر عجله برای خواندن آن [= قرآن‌] حرکت مده،</span>

##### 17

<span class="ayah">إِنَّ عَلَيْنَا جَمْعَهُۥ وَقُرْءَانَهُۥ</span>
<br><span class="ayah_translation">چرا که جمع‌کردن و خواندن آن بر عهده ماست!</span>

##### 18

<span class="ayah">فَإِذَا قَرَأْنَٰهُ فَٱتَّبِعْ قُرْءَانَهُۥ</span>
<br><span class="ayah_translation">پس هر گاه آن را خواندیم، از خواندن آن پیروی کن!</span>

##### 19

<span class="ayah">ثُمَّ إِنَّ عَلَيْنَا بَيَانَهُۥ</span>
<br><span class="ayah_translation">سپس بیان و (توضیح) آن (نیز) بر عهده ماست!</span>

##### 20

<span class="ayah">كَلَّا بَلْ تُحِبُّونَ ٱلْعَاجِلَةَ</span>
<br><span class="ayah_translation">چنین نیست که شما می‌پندارید (و دلایل معاد را کافی نمی‌دانید)؛ بلکه شما دنیای زودگذر را دوست دارید (و هوسرانی بی‌قید و شرط را)!</span>

##### 21

<span class="ayah">وَتَذَرُونَ ٱلْءَاخِرَةَ</span>
<br><span class="ayah_translation">و آخرت را رها می‌کنید!</span>

##### 22

<span class="ayah">وُجُوهٌۭ يَوْمَئِذٍۢ نَّاضِرَةٌ</span>
<br><span class="ayah_translation">(آری) در آن روز صورتهایی شاداب و مسرور است،</span>

##### 23

<span class="ayah">إِلَىٰ رَبِّهَا نَاظِرَةٌۭ</span>
<br><span class="ayah_translation">و به پروردگارش می‌نگرد!</span>

##### 24

<span class="ayah">وَوُجُوهٌۭ يَوْمَئِذٍۭ بَاسِرَةٌۭ</span>
<br><span class="ayah_translation">و در آن روز صورتهایی عبوس و در هم کشیده است،</span>

##### 25

<span class="ayah">تَظُنُّ أَن يُفْعَلَ بِهَا فَاقِرَةٌۭ</span>
<br><span class="ayah_translation">زیرا می‌داند عذابی در پیش دارد که پشت را در هم می‌شکند!</span>

##### 26

<span class="ayah">كَلَّآ إِذَا بَلَغَتِ ٱلتَّرَاقِىَ</span>
<br><span class="ayah_translation">چنین نیست (که انسان می‌پندارد! او ایمان نمی‌آورد) تا موقعی که جان به گلوگاهش رسد،</span>

##### 27

<span class="ayah">وَقِيلَ مَنْ ۜ رَاقٍۢ</span>
<br><span class="ayah_translation">و گفته شود: «آیا کسی هست که (این بیمار را از مرگ) نجات دهد؟!»</span>

##### 28

<span class="ayah">وَظَنَّ أَنَّهُ ٱلْفِرَاقُ</span>
<br><span class="ayah_translation">و به جدائی از دنیا یقین پیدا کند،</span>

##### 29

<span class="ayah">وَٱلْتَفَّتِ ٱلسَّاقُ بِٱلسَّاقِ</span>
<br><span class="ayah_translation">و ساق پاها (از سختی جان‌دادن) به هم بپیچد!</span>

##### 30

<span class="ayah">إِلَىٰ رَبِّكَ يَوْمَئِذٍ ٱلْمَسَاقُ</span>
<br><span class="ayah_translation">(آری) در آن روز مسیر همه بسوی (دادگاه) پروردگارت خواهد بود!</span>

##### 31

<span class="ayah">فَلَا صَدَّقَ وَلَا صَلَّىٰ</span>
<br><span class="ayah_translation">(در آن روز گفته می‌شود:) او هرگز ایمان نیاورد و نماز نخواند،</span>

##### 32

<span class="ayah">وَلَٰكِن كَذَّبَ وَتَوَلَّىٰ</span>
<br><span class="ayah_translation">بلکه تکذیب کرد و روی‌گردان شد،</span>

##### 33

<span class="ayah">ثُمَّ ذَهَبَ إِلَىٰٓ أَهْلِهِۦ يَتَمَطَّىٰٓ</span>
<br><span class="ayah_translation">سپس بسوی خانواده خود بازگشت در حالی که متکبّرانه قدم برمی‌داشت!</span>

##### 34

<span class="ayah">أَوْلَىٰ لَكَ فَأَوْلَىٰ</span>
<br><span class="ayah_translation">(با این اعمال) عذاب الهی برای تو شایسته‌تر است، شایسته‌تر!</span>

##### 35

<span class="ayah">ثُمَّ أَوْلَىٰ لَكَ فَأَوْلَىٰٓ</span>
<br><span class="ayah_translation">سپس عذاب الهی برای تو شایسته‌تر است، شایسته‌تر!</span>

##### 36

<span class="ayah">أَيَحْسَبُ ٱلْإِنسَٰنُ أَن يُتْرَكَ سُدًى</span>
<br><span class="ayah_translation">آیا انسان گمان می‌کند بی‌هدف رها می‌شود؟!</span>

##### 37

<span class="ayah">أَلَمْ يَكُ نُطْفَةًۭ مِّن مَّنِىٍّۢ يُمْنَىٰ</span>
<br><span class="ayah_translation">آیا او نطفه‌ای از منی که در رحم ریخته می‌شود نبود؟!</span>

##### 38

<span class="ayah">ثُمَّ كَانَ عَلَقَةًۭ فَخَلَقَ فَسَوَّىٰ</span>
<br><span class="ayah_translation">سپس بصورت خون بسته در آمد، و خداوند او را آفرید و موزون ساخت،</span>

##### 39

<span class="ayah">فَجَعَلَ مِنْهُ ٱلزَّوْجَيْنِ ٱلذَّكَرَ وَٱلْأُنثَىٰٓ</span>
<br><span class="ayah_translation">و از او دو زوج مرد و زن آفرید!</span>

##### 40

<span class="ayah">أَلَيْسَ ذَٰلِكَ بِقَٰدِرٍ عَلَىٰٓ أَن يُحْۦِىَ ٱلْمَوْتَىٰ</span>
<br><span class="ayah_translation">آیا چنین کسی قادر نیست که مردگان را زنده کند؟!</span>
